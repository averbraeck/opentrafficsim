93      /**
94       * Create a new ContourPlot.
95       * @param caption String; text to show above the plotting area
96       * @param xAxis Axis; the X (time) axis
97       * @param yAxis Axis; the Y axis
98       * @param redValue Double; contour value that will be rendered in Red
99       * @param yellowValue Double; contour value that will be rendered in Yellow
100      * @param greenValue Double; contour value that will be rendered in Green
101      * @param valueFormat String; format string for the contour values
102      * @param legendFormat String; format string for the captions in the color legend
103      * @param legendStep Double; increment between color legend entries
104      */
105     public ContourPlot(final String caption, final Axis xAxis, final Axis yAxis, final double redValue,
106             final double yellowValue, final double greenValue, final String valueFormat, final String legendFormat,
107             final double legendStep)
108     {
109         this.xAxis = xAxis;
110         this.yAxis = yAxis;
111         extendXRange(xAxis.getMaximumValue());
112         double[] boundaries = {redValue, yellowValue, greenValue};
113         this.chartPanel = new ChartPanel(createChart(caption, valueFormat, this, boundaries, legendFormat, legendStep));
114         final PointerHandler ph = new PointerHandler()
115         {
116             /** {@inheritDoc} */
117             @Override
118             void updateHint(final double domainValue, final double rangeValue)
119             {
120                 if (Double.isNaN(domainValue))
121                 {
122                     setStatusText(" ");
123                     return;
124                 }
125                 XYPlot plot = (XYPlot) getChartPanel().getChart().getPlot();
126                 XYZDataset dataset = (XYZDataset) plot.getDataset();
127                 String value = "";
128                 double roundedTime = domainValue;
129                 double roundedDistance = rangeValue;
130                 for (int item = dataset.getItemCount(0); --item >= 0;)
131                 {
132                     double x = dataset.getXValue(0, item);
133                     if (x + xAxis.getCurrentGranularity() / 2 < domainValue
134                             || x - xAxis.getCurrentGranularity() / 2 >= domainValue)
135                     {
136                         continue;
137                     }
138                     double y = dataset.getYValue(0, item);
139                     if (y + yAxis.getCurrentGranularity() / 2 < rangeValue
140                             || y - yAxis.getCurrentGranularity() / 2 >= rangeValue)
141                     {
142                         continue;
143                     }
144                     roundedTime = x;
145                     roundedDistance = y;
146                     double valueUnderMouse = dataset.getZValue(0, item);
147                     // System.out.println("Value under mouse is " + valueUnderMouse);
148                     if (Double.isNaN(valueUnderMouse))
149                     {
150                         break;
151                     }
152                     String format =
153                             ((ContinuousColorPaintScale) (((XYBlockRenderer) (plot.getRenderer(0))).getPaintScale()))
154                                     .getFormat();
155                     value = String.format(format, valueUnderMouse);
156                 }
157                 setStatusText(String.format("time %.0fs, distance %.0fm, %s", roundedTime, roundedDistance, value));
158             }
159 
160         };
161         this.chartPanel.addMouseMotionListener(ph);
162         this.chartPanel.addMouseListener(ph);
163         this.chartPanel.setMouseWheelEnabled(true);
164         add(this.chartPanel, BorderLayout.CENTER);
165         this.statusLabel = new JLabel(" ", SwingConstants.CENTER);
166         add(this.statusLabel, BorderLayout.SOUTH);
167         final JPopupMenu popupMenu = this.chartPanel.getPopupMenu();
168         popupMenu.insert(
169                 buildMenu("Distance granularity", "%.0f m", "setDistanceGranularity", yAxis.getGranularities(),
170                         yAxis.getCurrentGranularity()), 0);
171         popupMenu.insert(
172                 buildMenu("Time granularity", "%.0f s", "setTimeGranularity", xAxis.getGranularities(),
173                         xAxis.getCurrentGranularity()), 1);
174         reGraph();
175     }

186     /**
187      * Create a JMenu to let the user set the granularity of the XYBlockChart.
188      * @param caption String; caption for the new JMenu
189      * @param format String; format string for the values in the items under the new JMenu
190      * @param commandPrefix String; prefix for the actionCommand of the items under the new JMenu
191      * @param values double[]; array of values to be formatted using the format strings to yield the items under the new
192      *            JMenu
193      * @param currentValue double; the currently selected value (used to put the bullet on the correct item)
194      * @return JMenu with JRadioMenuItems for the values and a bullet on the currentValue item
195      */
196     private JMenu buildMenu(final String caption, final String format, final String commandPrefix,
197             final double[] values, final double currentValue)
198     {
199         final JMenu result = new JMenu(caption);
200         // Enlighten me: Do the menu items store a reference to the ButtonGroup so it won't get garbage collected?
201         final ButtonGroup group = new ButtonGroup();
202         for (double value : values)
203         {
204             final JRadioButtonMenuItem item = new JRadioButtonMenuItem(String.format(format, value));
205             item.setSelected(value == currentValue);
206             item.setActionCommand(commandPrefix + String.format(Locale.US, " %f", value));
207             item.addActionListener(this);
208             result.add(item);
209             group.add(item);
210         }
211         return result;
212     }

214     /**
215      * Create a XYBlockChart.
216      * @param caption String; text to show above the chart
217      * @param valueFormat String; format string used to render the value under the mouse in the status bar
218      * @param dataset XYZDataset with the values to render
219      * @param boundaries double[]; array of three boundary values corresponding to Red, Yellow and Green
220      * @param legendFormat String; the format string for captions in the legend
221      * @param legendStep value difference for successive colors in the legend. The first legend value displayed is equal
222      *            to the lowest value in boundaries.
223      * @return JFreeChart; the new XYBlockChart
224      */
225     private static JFreeChart createChart(final String caption, final String valueFormat, final XYZDataset dataset,
226             final double[] boundaries, final String legendFormat, final double legendStep)
227     {
228         final NumberAxis xAxis = new NumberAxis("\u2192 " + "time [s]");
229         xAxis.setLowerMargin(0.0);
230         xAxis.setUpperMargin(0.0);
231         final NumberAxis yAxis = new NumberAxis("\u2192 " + "Distance [m]");
232         yAxis.setAutoRangeIncludesZero(false);
233         yAxis.setLowerMargin(0.0);
234         yAxis.setUpperMargin(0.0);
235         yAxis.setStandardTickUnits(NumberAxis.createIntegerTickUnits());
236         XYBlockRenderer renderer = new XYBlockRenderer();
237         final Color[] colorValues = {Color.RED, Color.YELLOW, Color.GREEN};
238         final ContinuousColorPaintScale paintScale =
239                 new ContinuousColorPaintScale(valueFormat, boundaries, colorValues);
240         renderer.setPaintScale(paintScale);
241         final XYPlot plot = new XYPlot(dataset, xAxis, yAxis, renderer);
242         final LegendItemCollection legend = new LegendItemCollection();
243         for (int i = 0;; i++)
244         {
245             double value = paintScale.getLowerBound() + i * legendStep;
246             if (value > paintScale.getUpperBound())
247             {
248                 break;
249             }
250             legend.add(new LegendItem(String.format(legendFormat, value), paintScale.getPaint(value)));
251         }
252         legend.add(new LegendItem("No data", Color.BLACK));
253         plot.setFixedLegendItems(legend);
254         plot.setBackgroundPaint(Color.lightGray);
255         plot.setDomainGridlinePaint(Color.white);
256         plot.setRangeGridlinePaint(Color.white);
257         final JFreeChart chart = new JFreeChart(caption, plot);
258         chart.setBackgroundPaint(Color.white);
259         return chart;
260     }

262     /** {@inheritDoc} */
263     @Override
264     public final void actionPerformed(final ActionEvent actionEvent)
265     {
266         final String command = actionEvent.getActionCommand();
267         // System.out.println("command is \"" + command + "\"");
268         String[] fields = command.split("[ ]");
269         if (fields.length == 2)
270         {
271             final NumberFormat nf = NumberFormat.getInstance(Locale.US);
272             double value;
273             try
274             {
275                 value = nf.parse(fields[1]).doubleValue();
276             }
277             catch (ParseException e)
278             {
279                 throw new Error("Bad value: " + fields[1]);
280             }
281             if (fields[0].equalsIgnoreCase("setDistanceGranularity"))
282             {
283                 this.getYAxis().setCurrentGranularity(value);
284             }
285             else if (fields[0].equalsIgnoreCase("setTimeGranularity"))
286             {
287                 this.getXAxis().setCurrentGranularity(value);
288             }
289             else
290             {
291                 throw new Error("Unknown ActionEvent");
292             }
293             reGraph();
294         }
295         else
296         {
297             throw new Error("Unknown ActionEvent: " + command);
298         }
299     }

311         blockRenderer.setBlockWidth(this.getXAxis().getCurrentGranularity());

472     /**
473      * Add a fragment of a trajectory to this ContourPlot.
474      * @param car Car; the GTU that is being sampled (should be a GTU)
475      */
476     public final void addData(final Car car)
477     {
478         final DoubleScalar.Abs<TimeUnit> fromTime = car.getLastEvaluationTime();
479         final DoubleScalar.Abs<TimeUnit> toTime = car.getNextEvaluationTime();
480         if (toTime.getValueSI() > this.getXAxis().getMaximumValue().getValueSI())
481         {
482             extendXRange(toTime);
483             this.getXAxis().adjustMaximumValue(toTime);
484         }
485         if (toTime.getValueSI() <= fromTime.getValueSI()) // degenerate sample???
486         {
487             return;
488         }
489         /*-
490         System.out.println(String.format("addData: fromTime=%.1f, toTime=%.1f, fromDist=%.2f, toDist=%.2f", fromTime
491                 .getValueSI(), toTime.getValueSI(), car.position(fromTime).getValueSI(), car.position(toTime)
492                 .getValueSI()));
493          */
494         // The "relative" values are "counting" distance or time in the minimum bin size unit
495         final double relativeFromDistance =
496                 (car.getPosition(fromTime).getValueSI() - this.getYAxis().getMinimumValue().getValueSI())
497                         / this.getYAxis().getGranularities()[0];
498         final double relativeToDistance =
499                 (car.getPosition(toTime).getValueSI() - this.getYAxis().getMinimumValue().getValueSI())
500                         / this.getYAxis().getGranularities()[0];
501         double relativeFromTime =
502                 (fromTime.getValueSI() - this.getXAxis().getMinimumValue().getValueSI())
503                         / this.getXAxis().getGranularities()[0];
504         final double relativeToTime =
505                 (toTime.getValueSI() - this.getXAxis().getMinimumValue().getValueSI())
506                         / this.getXAxis().getGranularities()[0];
507         final int fromTimeBin = (int) Math.floor(relativeFromTime);
508         final int toTimeBin = (int) Math.floor(relativeToTime) + 1;
509         double relativeMeanSpeed = (relativeToDistance - relativeFromDistance) / (relativeToTime - relativeFromTime);
510         // The code for acceleration assumes that acceleration is constant (which is correct for IDM+, but may be
511         // wrong for other car following algorithms).
512         double acceleration = car.getAcceleration(car.getLastEvaluationTime()).getValueSI();
513         for (int timeBin = fromTimeBin; timeBin < toTimeBin; timeBin++)
514         {
515             if (timeBin < 0)
516             {
517                 continue;
518             }
519             double binEndTime = timeBin + 1;
520             if (binEndTime > relativeToTime)
521             {
522                 binEndTime = relativeToTime;
523             }
524             if (binEndTime <= relativeFromTime)
525             {
526                 continue; // no time spent in this timeBin
527             }
528             double binDistanceStart =
529                     (car.getPosition(
530                             new DoubleScalar.Abs<TimeUnit>(relativeFromTime * this.getXAxis().getGranularities()[0],
531                                     TimeUnit.SECOND)).getValueSI() - this.getYAxis().getMinimumValue().getValueSI())
532                             / this.getYAxis().getGranularities()[0];
533             double binDistanceEnd =
534                     (car.getPosition(
535                             new DoubleScalar.Abs<TimeUnit>(binEndTime * this.getXAxis().getGranularities()[0],
536                                     TimeUnit.SECOND)).getValueSI() - this.getYAxis().getMinimumValue().getValueSI())
537                             / this.getYAxis().getGranularities()[0];
538 
539             // Compute the time in each distanceBin
540             for (int distanceBin = (int) Math.floor(binDistanceStart); distanceBin <= binDistanceEnd; distanceBin++)
541             {
542                 double relativeDuration = 1;
543                 if (relativeFromTime > timeBin)
544                 {
545                     relativeDuration -= relativeFromTime - timeBin;
546                 }
547                 if (distanceBin == (int) Math.floor(binDistanceEnd))
548                 {
549                     // This GTU does not move out of this distanceBin before the binEndTime
550                     if (binEndTime < timeBin + 1)
551                     {
552                         relativeDuration -= timeBin + 1 - binEndTime;
553                     }
554                 }
555                 else
556                 {
557                     // This GTU moves out of this distanceBin before the binEndTime
558                     // Interpolate the time when this GTU crosses into the next distanceBin
559                     // Using f.i. Newton-Rhaphson interpolation would yield a slightly more precise result...
560                     double timeToBinBoundary = (distanceBin + 1 - binDistanceStart) / relativeMeanSpeed;
561                     double endTime = relativeFromTime + timeToBinBoundary;
562                     relativeDuration -= timeBin + 1 - endTime;
563                 }
564                 final double duration = relativeDuration * this.getXAxis().getGranularities()[0];
565                 final double distance = duration * relativeMeanSpeed * this.getYAxis().getGranularities()[0];
566                 /*-
567                 System.out.println(String.format("tb=%d, db=%d, t=%.2f, d=%.2f", timeBin, distanceBin, duration,
568                         distance));
569                  */
570                 incrementBinData(timeBin, distanceBin, duration, distance, acceleration);
571                 relativeFromTime += relativeDuration;
572                 binDistanceStart = distanceBin + 1;
573             }
574             relativeFromTime = timeBin + 1;
575         }
576 
577     }

597     /** {@inheritDoc} */
598     @Override
599     public final double getZValue(final int series, final int item)
600     {
601         final int timeBinGroup = xAxisBin(item);
602         final int distanceBinGroup = yAxisBin(item);
603         // System.out.println(String.format("getZValue(s=%d, i=%d) -> tbg=%d, dbg=%d", series, item, timeBinGroup,
604         // distanceBinGroup));
605         final int timeGroupSize =
606                 (int) (this.getXAxis().getCurrentGranularity() / this.getXAxis().getGranularities()[0]);
607         final int firstTimeBin = timeBinGroup * timeGroupSize;
608         final int distanceGroupSize =
609                 (int) (this.getYAxis().getCurrentGranularity() / this.getYAxis().getGranularities()[0]);
610         final int firstDistanceBin = distanceBinGroup * distanceGroupSize;
611         final int endTimeBin = Math.min(firstTimeBin + timeGroupSize, this.getXAxis().getBinCount());
612         final int endDistanceBin = Math.min(firstDistanceBin + distanceGroupSize, this.getYAxis().getBinCount());
613         return computeZValue(firstTimeBin, endTimeBin, firstDistanceBin, endDistanceBin);
614     }
