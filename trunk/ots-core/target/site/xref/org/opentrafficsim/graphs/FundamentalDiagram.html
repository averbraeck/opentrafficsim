154     /**
155      * Graph a Fundamental Diagram.
156      * @param caption String; the caption shown above the graphing area.
157      * @param numberOfLanes Integer; the number of lanes covered by the detector that generates the data for this
158      *            Fundamental diagram
159      * @param aggregationTime DoubleScalarRel&lt;TimeUnit&gt;; the aggregation of the detector that generates the data
160      *            for this Fundamental diagram
161      * @param position DoubleScalarAbs&lt;LengthUnit&gt;; position of the detector (FIXME: should be a LOT more general)
162      */
163     public FundamentalDiagram(final String caption, final int numberOfLanes,
164             final DoubleScalar.Rel<TimeUnit> aggregationTime, final DoubleScalar.Abs<LengthUnit> position)
165     {
166         if (numberOfLanes <= 0)
167         {
168             throw new Error("Number of lanes must be > 0 (got " + numberOfLanes + ")");
169         }
170         if (aggregationTime.getValueSI() <= 0)
171         {
172             throw new Error("Aggregation time must be > 0 (got " + aggregationTime + ")");
173         }
174         this.aggregationTime = aggregationTime;
175         this.sampleSets = new ArrayList<ArrayList<Sample>>(numberOfLanes);
176         for (int i = 0; i < numberOfLanes; i++)
177         {
178             this.sampleSets.add(new ArrayList<Sample>());
179         }
180         this.caption = caption;
181         this.position = new DoubleScalar.Abs<LengthUnit>(position);
182         ChartFactory.setChartTheme(new StandardChartTheme("JFree/Shadow", false));
183         this.chartPanel =
184                 ChartFactory.createXYLineChart(this.caption, "", "", this, PlotOrientation.VERTICAL, false, false,
185                         false);
186         final XYLineAndShapeRenderer renderer = (XYLineAndShapeRenderer) this.chartPanel.getXYPlot().getRenderer();
187         renderer.setBaseLinesVisible(true);
188         renderer.setBaseShapesVisible(true);
189         renderer.setBaseItemLabelGenerator(new XYItemLabelGenerator()
190         {
191             @Override
192             public String generateLabel(final XYDataset dataset, final int series, final int item)
193             {
194                 return String.format("%.0fs", item * aggregationTime.getValueSI());
195             }
196         });
197         renderer.setBaseItemLabelsVisible(true);
198         final ChartPanel cp = new ChartPanel(this.chartPanel);
199         PointerHandler ph = new PointerHandler()
200         {
201             /** {@inheritDoc} */
202             @Override
203             void updateHint(final double domainValue, final double rangeValue)
204             {
205                 if (Double.isNaN(domainValue))
206                 {
207                     setStatusText(" ");
208                     return;
209                 }
210                 String s1 = String.format(getXAxisFormat(), domainValue);
211                 String s2 = String.format(getYAxisFormat(), rangeValue);
212                 setStatusText(s1 + ", " + s2);
213             }
214 
215         };
216         cp.addMouseMotionListener(ph);
217         cp.addMouseListener(ph);
218         cp.setMouseWheelEnabled(true);
219         final JMenu subMenu = new JMenu("Set layout");
220         final ButtonGroup group = new ButtonGroup();
221         final JRadioButtonMenuItem defaultItem = addMenuItem(subMenu, group, getDensityAxis(), this.flowAxis, true);
222         addMenuItem(subMenu, group, this.flowAxis, this.speedAxis, false);
223         addMenuItem(subMenu, group, this.densityAxis, this.speedAxis, false);
224         actionPerformed(new ActionEvent(this, 0, defaultItem.getActionCommand()));
225         final JPopupMenu popupMenu = cp.getPopupMenu();
226         popupMenu.insert(subMenu, 0);
227         this.add(cp, BorderLayout.CENTER);
228         this.statusLabel = new JLabel(" ", SwingConstants.CENTER);
229         this.add(this.statusLabel, BorderLayout.SOUTH);
230     }

273     /**
274      * Add the effect of one passing car to this Fundamental Diagram.
275      * @param lane Integer; the lane on which the car passes
276      * @param car Car; the car that passes FIXME replace Car by GTU
277      * @param detectionTime DoubleScalarAbs&lt;TimeUnit&gt;; the time at which the GTU passes the detector
278      */
279     public final void addData(final int lane, final Car car, final DoubleScalar.Abs<TimeUnit> detectionTime)
280     {
281         ArrayList<Sample> laneData = this.sampleSets.get(lane);
282         // Figure out the time bin
283         final int timeBin = (int) Math.floor(detectionTime.getValueSI() / this.aggregationTime.getValueSI());
284         // Extend storage if needed
285         while (timeBin >= laneData.size())
286         {
287             laneData.add(new Sample());
288         }
289         Sample sample = laneData.get(timeBin);
290         sample.addData(car.getVelocity(detectionTime));
291     }

314         configureAxis(numberAxis, this.yAxis);

317         notifyListeners(new DatasetChangeEvent(this, null)); // This guess work actually works!

400     /**
401      * Retrieve a value from the recorded samples.
402      * @param lane Integer; the lane
403      * @param item Integer; the rank number of the sample
404      * @param axis Axis; the axis that determines which quantity to retrieve
405      * @return Double; the requested value, or Double.NaN if the sample does not (yet) exist
406      */
407     private Double getSample(final int lane, final int item, final Axis axis)
408     {
409         if (lane < 0 || lane >= this.sampleSets.size() || item < 0)
410         {
411             return Double.NaN;
412         }
413         ArrayList<Sample> laneDetections = this.sampleSets.get(lane);
414         if (item >= laneDetections.size())
415         {
416             return Double.NaN;
417         }
418         double result = laneDetections.get(item).getValue(axis);
419         /*-
420         System.out.println(String.format("getSample(lane=%d, item=%d, axis=%s) returns %f", lane, item, axis.name,
421                 result));
422          */
423         return result;
424     }

483     class Sample
484     {
485         /** Harmonic mean speed observed during this sample [m/s]. */
486         private double harmonicMeanSpeed;
487 
488         /** Flow observed during this sample [veh/s]. */
489         private double flow;
490 
491         /**
492          * Retrieve a value stored in this Sample.
493          * @param axis Axis; the axis along which the data is requested
494          * @return double; the retrieved value
495          */
496         public double getValue(final Axis axis)
497         {
498             if (axis == getDensityAxis())
499             {
500                 return this.flow * 3600 / getAggregationTime().getValueSI() / this.harmonicMeanSpeed;
501             }
502             else if (axis == getFlowAxis())
503             {
504                 return this.flow * 3600 / getAggregationTime().getValueSI();
505             }
506             else if (axis == getSpeedAxis())
507             {
508                 return this.harmonicMeanSpeed * 3600 / 1000;
509             }
510             else
511             {
512                 throw new Error("Sample.getValue: Can not identify axis");
513             }
514         }
515 
516         /**
517          * Add one Car detection to this Sample.
518          * @param speed DoubleScalar.Rel&lt;SpeedUnit&gt;; the detected speed
519          */
520         public void addData(final DoubleScalar.Rel<SpeedUnit> speed)
521         {
522             double sumReciprocalSpeeds = 0;
523             if (this.flow > 0)
524             {
525                 sumReciprocalSpeeds = this.flow / this.harmonicMeanSpeed;
526             }
527             this.flow += 1;
528             sumReciprocalSpeeds += 1d / speed.getValueSI();
529             this.harmonicMeanSpeed = this.flow / sumReciprocalSpeeds;
530         }
531     }

533     /** {@inheritDoc} */
534     @Override
535     public final void actionPerformed(final ActionEvent actionEvent)
536     {
537         final String command = actionEvent.getActionCommand();
538         // System.out.println("command is \"" + command + "\"");
539         final String[] fields = command.split("[/]");
540         if (fields.length == 2)
541         {
542             for (String field : fields)
543             {
544                 if (field.equalsIgnoreCase(this.densityAxis.getShortName()))
545                 {
546                     if (field == fields[0])
547                     {
548                         this.yAxis = this.densityAxis;
549                     }
550                     else
551                     {
552                         this.xAxis = this.densityAxis;
553                     }
554                 }
555                 else if (field.equalsIgnoreCase(this.flowAxis.getShortName()))
556                 {
557                     if (field == fields[0])
558                     {
559                         this.yAxis = this.flowAxis;
560                     }
561                     else
562                     {
563                         this.xAxis = this.flowAxis;
564                     }
565                 }
566                 else if (field.equalsIgnoreCase(this.speedAxis.getShortName()))
567                 {
568                     if (field == fields[0])
569                     {
570                         this.yAxis = this.speedAxis;
571                     }
572                     else
573                     {
574                         this.xAxis = this.speedAxis;
575                     }
576                 }
577                 else
578                 {
579                     throw new Error("Cannot find axis name: " + field);
580                 }
581             }
582             reGraph();
583         }
584         else
585         {
586             throw new Error("Unknown ActionEvent");
587         }
588     }
