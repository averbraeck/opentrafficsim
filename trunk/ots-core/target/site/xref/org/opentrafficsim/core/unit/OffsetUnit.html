3   import org.opentrafficsim.core.unit.unitsystem.UnitSystem;
4   
5   /**
6    * The OffsetUnit provides a unit where scales can have an offset, such as the temperature scale. Internally, all units
7    * are internally <i>stored</i> as a standard unit with an offset and a conversion factor. This means that e.g., Kelvin
8    * is stored with offset 0.0 and conversion factor 1.0, whereas degree Celsius is stored with offset -273.15 and
9    * conversion factor 1.0. This means that if we have a Temperature, it is stored in Kelvins, and if we want to display
10   * it in degree Celsius, we have to <i>divide</i> by the conversion factor and <i>subtract</i> the offset.
11   * <p>
12   * Copyright (c) 2014 Delft University of Technology, PO Box 5, 2600 AA, Delft, the Netherlands. All rights reserved. <br>
13   * BSD-style license. See <a href="http://opentrafficsim.org/node/13">OpenTrafficSim License</a>.
14   * <p>
15   * @version Jun 5, 2014 <br>
16   * @author <a href="http://www.tbm.tudelft.nl/averbraeck">Alexander Verbraeck</a>
17   * @param <U> the unit type
18   */
19  public abstract class OffsetUnit<U extends Unit<U>> extends Unit<U>
20  {
21      /** */
22      private static final long serialVersionUID = 20140607L;
23  
24      /** the offset that has to be taken into account for conversions. */
25      private final double offsetToStandardUnit;
26  
27      /**
28       * Build a standard unit.
29       * @param nameKey the key to the locale file for the long name of the unit
30       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
31       * @param unitSystem the unit system, e.g. SI or Imperial
32       */
33      public OffsetUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem)
34      {
35          super(nameKey, abbreviationKey, unitSystem, true);
36          this.offsetToStandardUnit = 0.0;
37      }
38  
39      /**
40       * Build an offset unit with a conversion factor and offset to another unit.
41       * @param nameKey the key to the locale file for the long name of the unit
42       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
43       * @param unitSystem the unit system, e.g. SI or Imperial
44       * @param referenceUnit the unit to convert to
45       * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
46       *            reference unit
47       * @param offsetToStandardUnit the offset to add to convert to the standard (e.g., SI) unit
48       */
49      public OffsetUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
50              final U referenceUnit, final double conversionFactorToReferenceUnit, final double offsetToStandardUnit)
51      {
52          super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, true);
53          this.offsetToStandardUnit = offsetToStandardUnit;
54      }
55  
56      /**
57       * @return offset to the standard unit. E.g., -273.15 to go from degrees Celsius to Kelvin
58       */
59      public final double getOffsetToStandardUnit()
60      {
61          return this.offsetToStandardUnit;
62      }
63  
64  }
