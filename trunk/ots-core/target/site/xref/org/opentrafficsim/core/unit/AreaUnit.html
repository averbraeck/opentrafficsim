18  public class AreaUnit extends Unit<AreaUnit>
19  {
20      /** */
21      private static final long serialVersionUID = 20140607L;
22  
23      /** the unit of length for the area unit, e.g., meter. */
24      private final LengthUnit lengthUnit;
25  
26      /** m^2. */
27      public static final AreaUnit SQUARE_METER;
28  
29      /** km^2. */
30      public static final AreaUnit SQUARE_KM;
31  
32      /** cm^2. */
33      public static final AreaUnit SQUARE_CENTIMETER;
34  
35      /** cm^2. */
36      public static final AreaUnit SQUARE_MILLIMETER;
37  
38      /** are. */
39      public static final AreaUnit ARE;
40  
41      /** hectare. */
42      public static final AreaUnit HECTARE;
43  
44      /** mile^2. */
45      public static final AreaUnit SQUARE_MILE;
46  
47      /** ft^2. */
48      public static final AreaUnit SQUARE_FOOT;
49  
50      /** in^2. */
51      public static final AreaUnit SQUARE_INCH;
52  
53      /** yd^2. */
54      public static final AreaUnit SQUARE_YARD;
55  
56      /** acre (international). */
57      public static final AreaUnit ACRE;
58  
59      static
60      {
61          SQUARE_METER = new AreaUnit(LengthUnit.METER, "AreaUnit.square_meter", "AreaUnit.m^2", SI_DERIVED);
62          SQUARE_KM = new AreaUnit(LengthUnit.KILOMETER, "AreaUnit.square_kilometer", "AreaUnit.km^2", SI_DERIVED);
63          SQUARE_CENTIMETER =
64                  new AreaUnit(LengthUnit.CENTIMETER, "AreaUnit.square_centimeter", "AreaUnit.cm^2", SI_DERIVED);
65          SQUARE_MILLIMETER =
66                  new AreaUnit(LengthUnit.MILLIMETER, "AreaUnit.square_millimeter", "AreaUnit.mm^2", SI_DERIVED);
67          ARE = new AreaUnit("AreaUnit.are", "AreaUnit.a", OTHER, SQUARE_METER, 100.0);
68          HECTARE = new AreaUnit("AreaUnit.hectare", "AreaUnit.ha", OTHER, ARE, 100.0);
69          SQUARE_MILE = new AreaUnit(LengthUnit.MILE, "AreaUnit.square_mile", "AreaUnit.mi^2", IMPERIAL);
70          SQUARE_FOOT = new AreaUnit(LengthUnit.FOOT, "AreaUnit.square_foot", "AreaUnit.ft^2", IMPERIAL);
71          SQUARE_INCH = new AreaUnit(LengthUnit.INCH, "AreaUnit.square_inch", "AreaUnit.in^2", IMPERIAL);
72          SQUARE_YARD = new AreaUnit(LengthUnit.YARD, "AreaUnit.square_yard", "AreaUnit.yd^2", IMPERIAL);
73          ACRE = new AreaUnit("AreaUnit.acre", "AreaUnit.ac", IMPERIAL, SQUARE_YARD, 4840.0);
74      }
75  
76      /**
77       * Define area unit based on length.
78       * @param lengthUnit the unit of length for the area unit, e.g., meter
79       * @param nameKey the key to the locale file for the long name of the unit
80       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
81       * @param unitSystem the unit system, e.g. SI or Imperial
82       */
83      public AreaUnit(final LengthUnit lengthUnit, final String nameKey, final String abbreviationKey,
84              final UnitSystem unitSystem)
85      {
86          super(nameKey, abbreviationKey, unitSystem, SQUARE_METER, lengthUnit.getConversionFactorToStandardUnit()
87                  * lengthUnit.getConversionFactorToStandardUnit(), true);
88          this.lengthUnit = lengthUnit;
89      }
90  
91      /**
92       * This constructor constructs a unit out of another defined unit, e.g. an are is 100 m^2.
93       * @param nameKey the key to the locale file for the long name of the unit
94       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
95       * @param unitSystem the unit system, e.g. SI or Imperial
96       * @param referenceUnit the unit to convert to
97       * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
98       *            reference unit
99       */
100     public AreaUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
101             final AreaUnit referenceUnit, final double conversionFactorToReferenceUnit)
102     {
103         super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, true);
104         this.lengthUnit = referenceUnit.getLengthUnit();
105     }
106 
107     /**
108      * @return lengthUnit
109      */
110     public final LengthUnit getLengthUnit()
111     {
112         return this.lengthUnit;
113     }
114 
115     /** {@inheritDoc} */
116     @Override
117     public final AreaUnit getStandardUnit()
118     {
119         return SQUARE_METER;
120     }
121 
122     /** {@inheritDoc} */
123     @Override
124     public final String getSICoefficientsString()
125     {
126         return "m2";
127     }
128 
129 }
