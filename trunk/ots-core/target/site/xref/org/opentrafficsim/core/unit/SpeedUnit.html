1   package org.opentrafficsim.core.unit;
2   
3   import static org.opentrafficsim.core.unit.unitsystem.UnitSystem.IMPERIAL;
4   import static org.opentrafficsim.core.unit.unitsystem.UnitSystem.SI_DERIVED;
5   
6   import org.opentrafficsim.core.unit.unitsystem.UnitSystem;
7   
8   /**
9    * According to <a href="http://en.wikipedia.org/wiki/Velocity">Wikipedia</a>: Speed describes only how fast an object
10   * is moving, whereas velocity gives both how fast and in what direction the object is moving.
11   * <p>

14   * <p>
15   * @version May 15, 2014 <br>
16   * @author <a href="http://www.tbm.tudelft.nl/averbraeck">Alexander Verbraeck</a>
17   */
18  public class SpeedUnit extends Unit<SpeedUnit>
19  {
20      /** */
21      private static final long serialVersionUID = 20140607L;
22  
23      /** the unit of length for the speed unit, e.g., meter. */
24      private final LengthUnit lengthUnit;
25  
26      /** the unit of time for the speed unit, e.g., second. */
27      private final TimeUnit timeUnit;
28  
29      /** m/s. */
30      public static final SpeedUnit METER_PER_SECOND;
31  
32      /** km/h. */
33      public static final SpeedUnit KM_PER_HOUR;
34  
35      /** mile/h. */
36      public static final SpeedUnit MILE_PER_HOUR;
37  
38      /** ft/s. */
39      public static final SpeedUnit FOOT_PER_SECOND;
40  
41      /** knot. */
42      public static final SpeedUnit KNOT;
43  
44      static
45      {
46          METER_PER_SECOND =
47                  new SpeedUnit(LengthUnit.METER, TimeUnit.SECOND, "SpeedUnit.meter_per_second", "SpeedUnit.m/s",
48                          SI_DERIVED);
49          KM_PER_HOUR =
50                  new SpeedUnit(LengthUnit.KILOMETER, TimeUnit.HOUR, "SpeedUnit.kilometer_per_hour", "SpeedUnit.km/h",
51                          SI_DERIVED);
52          MILE_PER_HOUR =
53                  new SpeedUnit(LengthUnit.MILE, TimeUnit.HOUR, "SpeedUnit.mile_per_hour", "SpeedUnit.mph", IMPERIAL);
54          FOOT_PER_SECOND =
55                  new SpeedUnit(LengthUnit.FOOT, TimeUnit.SECOND, "SpeedUnit.foot_per_second", "SpeedUnit.fps", IMPERIAL);
56          KNOT = new SpeedUnit(LengthUnit.NAUTICAL_MILE, TimeUnit.HOUR, "SpeedUnit.knot", "SpeedUnit.kt", IMPERIAL);
57      }
58  
59      /**
60       * Build a speed unit from a length unit and a time unit.
61       * @param lengthUnit the unit of length for the speed unit, e.g., meter
62       * @param timeUnit the unit of time for the speed unit, e.g., second
63       * @param nameKey the key to the locale file for the long name of the unit
64       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
65       * @param unitSystem the unit system, e.g. SI or Imperial
66       */
67      public SpeedUnit(final LengthUnit lengthUnit, final TimeUnit timeUnit, final String nameKey,
68              final String abbreviationKey, final UnitSystem unitSystem)
69      {
70          super(nameKey, abbreviationKey, unitSystem, METER_PER_SECOND, lengthUnit.getConversionFactorToStandardUnit()
71                  / timeUnit.getConversionFactorToStandardUnit(), true);
72          this.lengthUnit = lengthUnit;
73          this.timeUnit = timeUnit;
74      }
75  
76      /**
77       * Build a speed unit based on another speed unit.
78       * @param nameKey the key to the locale file for the long name of the unit
79       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
80       * @param unitSystem the unit system, e.g. SI or Imperial
81       * @param referenceUnit the unit to convert to
82       * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
83       *            reference unit
84       */
85      public SpeedUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
86              final SpeedUnit referenceUnit, final double conversionFactorToReferenceUnit)
87      {
88          super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, true);
89          this.lengthUnit = referenceUnit.getLengthUnit();
90          this.timeUnit = referenceUnit.getTimeUnit();
91      }
92  
93      /**
94       * @return lengthUnit
95       */
96      public final LengthUnit getLengthUnit()
97      {
98          return this.lengthUnit;
99      }
100 
101     /**
102      * @return timeUnit
103      */
104     public final TimeUnit getTimeUnit()
105     {
106         return this.timeUnit;
107     }
108 
109     /** {@inheritDoc} */
110     @Override
111     public final SpeedUnit getStandardUnit()
112     {
113         return METER_PER_SECOND;
114     }
115 
116     /** {@inheritDoc} */
117     @Override
118     public final String getSICoefficientsString()
119     {
120         return "m/s";
121     }
122 
123 }
