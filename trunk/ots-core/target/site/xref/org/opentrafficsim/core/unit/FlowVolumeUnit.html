130     /**
131      * Create a flow-volumeunit based on length (cubed) per time unit.
132      * @param lengthUnit the unit of length for the flow unit, e.g., meter
133      * @param timeUnit the unit of time for the flow unit, e.g., second
134      * @param nameKey the key to the locale file for the long name of the unit
135      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
136      * @param unitSystem the unit system, e.g. SI or Imperial
137      */
138     public FlowVolumeUnit(final LengthUnit lengthUnit, final TimeUnit timeUnit, final String nameKey,
139             final String abbreviationKey, final UnitSystem unitSystem)
140     {
141         super(nameKey, abbreviationKey, unitSystem, CUBIC_METER_PER_SECOND, Math.pow(
142                 lengthUnit.getConversionFactorToStandardUnit(), 3.0)
143                 / timeUnit.getConversionFactorToStandardUnit(), true);
144         this.lengthUnit = lengthUnit;
145         this.timeUnit = timeUnit;
146     }
147 
148     /**
149      * Create a flow-volumeunit based as a volume unit per time unit.
150      * @param volumeUnit the unit of volume for the flow unit, e.g., cubic meter
151      * @param timeUnit the unit of time for the flow unit, e.g., second
152      * @param nameKey the key to the locale file for the long name of the unit
153      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
154      * @param unitSystem the unit system, e.g. SI or Imperial
155      */
156     public FlowVolumeUnit(final VolumeUnit volumeUnit, final TimeUnit timeUnit, final String nameKey,
157             final String abbreviationKey, final UnitSystem unitSystem)
158     {
159         super(nameKey, abbreviationKey, unitSystem, CUBIC_METER_PER_SECOND, volumeUnit
160                 .getConversionFactorToStandardUnit() / timeUnit.getConversionFactorToStandardUnit(), true);
161         this.lengthUnit = volumeUnit.getLengthUnit();
162         this.timeUnit = timeUnit;
163     }
164 
165     /**
166      * Create a flow-volumeunit based on another flow-volumeunit.
167      * @param nameKey the key to the locale file for the long name of the unit
168      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
169      * @param unitSystem the unit system, e.g. SI or Imperial
170      * @param referenceUnit the unit to convert to
171      * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
172      *            reference unit
173      */
174     public FlowVolumeUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
175             final FlowVolumeUnit referenceUnit, final double conversionFactorToReferenceUnit)
176     {
177         super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, true);
178         this.lengthUnit = referenceUnit.getLengthUnit();
179         this.timeUnit = referenceUnit.getTimeUnit();
180     }
