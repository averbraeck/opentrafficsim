7   /**
8    * The Node is a point with an id. It is used in the network to connect Links.

16   * @author <a href="http://www.citg.tudelft.nl">Guus Tamminga</a>
17   * @param <ID> the ID type.
18   * @param <P> the point type, e.g., com.vividsolutions.jts.geom.Point, DirectedPoint. 
19   */
20  public abstract class AbstractNode<ID, P> implements LocatableInterface, Serializable
21  {
22      /** */
23      private static final long serialVersionUID = 20140920L;
24  
25      /** the node id. */
26      private ID id;
27  
28      /** the point. */
29      private P point;
30  
31      /**
32       * Construction of a Node.
33       * @param id the id of the Node.
34       * @param point the point with usually an x and y setting.
35       */
36  
37      public AbstractNode(final ID id, final P point)
38      {
39          this.id = id;
40          this.point = point;
41      }
42  
43      /**
44       * @return node id.
45       */
46      public final ID getId()
47      {
48          return this.id;
49      }
50  
51      /**
52       * @return point.
53       */
54      public final P getPoint()
55      {
56          return this.point;
57      }
58  
59  }
