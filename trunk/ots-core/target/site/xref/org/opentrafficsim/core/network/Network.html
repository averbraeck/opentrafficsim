9   /**
10   * A Network consists of a set of links. Each link has, in its turn, a start node and an end node. In addition, a network can be
11   * an (expanded) node as well. An example is shown below:
12   * 
13   * <pre>
14   *            |
15   *     -------O--------
16   *            |
17   * </pre>
18   * 
19   * can be expanded into:
20   * 
21   * <pre>
22   *            |
23   *            A
24   *           /|\
25   *          / | \
26   *    -----B--C--D-----
27   *          \ | /
28   *           \|/
29   *            E
30   *            |
31   * </pre>
32   * 
33   * Node O in the example is expanded into the subnetwork consisting of nodes A, B, C, D, and E, and links AB, AC, AD, BC, CD,
34   * BE, CE, and DE. It also means that when node expansion takes place, the links to node O have to be replaced. In the example
35   * below:
36   * 
37   * <pre>
38   *            X
39   *            |
40   *     Y------O-------Z
41   *            |
42   *            W
43   * </pre>
44   * 
45   * can be expanded into:
46   * 
47   * <pre>
48   *            X
49   *            |
50   *            A
51   *           /|\
52   *          / | \
53   *    Y----B--C--D----Z
54   *          \ | /
55   *           \|/
56   *            E
57   *            |
58   *            W
59   * </pre>
60   * 
61   * the node XO is replaced by XA, YO is replaced by YB, OZ is replaced by DZ, and OW is replaced by EW in the network. The
62   * reverse takes place when we do node collapse.
63   * <p>

65   * BSD-style license. See <a href="http://opentrafficsim.org/node/13">OpenTrafficSim License</a>.
66   * <p>
67   * @version Aug 19, 2014 <br>
68   * @author <a href="http://www.tbm.tudelft.nl/averbraeck">Alexander Verbraeck</a>
69   * @author <a href="http://www.tudelft.nl/pknoppers">Peter Knoppers</a>
70   * @author <a href="http://www.citg.tudelft.nl">Guus Tamminga</a>

109     /**
110      * Construction of a network with an initial set of links, and an expansion node.
111      * @param id the network id.
112      * @param collection the initial collection of links.
113      * @param expansionNode Node of which this network is an expansion.
114      * @throws NetworkException when expansion node is part of the initial collection.
115      */
116     public Network(final ID id, final Collection<? extends L> collection, final AbstractNode<?, ?> expansionNode)
117             throws NetworkException
118     {
119         super(collection);
120         this.id = id;
121         if (collection.contains(expansionNode))
122             throw new NetworkException("Creating Network " + this.id + " with initial collection. Expansion node "
123                     + expansionNode.toString() + " is part of the initial collection");
124         this.expansionOfNode = expansionNode;
125     }

170     /**
171      * @param node
172      * @return true or false
173      */
174     public boolean isInNetwork(AbstractNode<?, ?> node)
175     {
176 
177         if (this.nodeSet.contains(node))
178         {
179             return true;
180         }
181         else
182         {
183             for (AbstractNode<?, ?> n : this.nodeSet)
184             {
185                 if (n instanceof AbstractExpansionNode)
186                 {
187                     if (((AbstractExpansionNode<?, ?>) n).getNetwork().isInNetwork(node))
188                     {
189                         return true;
190                     }
191                 }
192             }
193         }
194 
195         return false;
196     }

198     /**
199      * @param addThis
200      * @return true or false
201      * @throws NetworkException
202      */
203     public boolean addNode(AbstractNode<?, ?> addThis) throws NetworkException
204     {
205         if (isInNetwork(addThis))
206         {
207             throw new NetworkException("Adding Node " + addThis.getId().toString() + ". This Node is  already in the Set");
208         }
209         else
210         {
211             this.nodeSet.add(addThis);
212             return true;
213         }
214 
215     }

217     /**
218      * @param node
219      * @return boolean
220      */
221     public boolean isInNetworkLevel(AbstractNode<?, ?> node)
222     {
223         if (this.nodeSet.contains(node))
224         {
225             return true;
226         }
227         else
228         {
229             return false;
230         }
231     }

233     /**
234      * @param node
235      * @return network
236      * @throws NetworkException
237      */
238     public Network<?, ?> getSubNetworkConsistNode(AbstractNode<?, ?> node) throws NetworkException
239     {
240         if (isInNetwork(node))
241         {
242             if (isInNetworkLevel(node))
243             {
244                 return this;
245             }
246             else
247             {
248                 for (AbstractNode<?, ?> n : this.nodeSet)
249                 {
250                     if (n instanceof AbstractExpansionNode)
251                     {
252                         if (((AbstractExpansionNode<?, ?>) n).getNetwork().isInNetworkLevel(node))
253                         {
254                             return getSubNetworkConsistNode(node);
255                         }
256                     }
257                 }
258             }
259         }
260         else
261         {
262             throw new NetworkException("The network does not contain the Node" + node.getId().toString() + ".");
263         }
264         return null;
265     }
266 
267     /**
268      * @param deleteThis
269      * @return boolean
270      * @throws NetworkException
271      */
272     public boolean deleteNode(AbstractNode<?, ?> deleteThis) throws NetworkException
273     {
274         if (isInNetwork(deleteThis))
275         {
276             if (isInNetworkLevel(deleteThis))
277             {
278                 this.nodeSet.remove(deleteThis);
279                 return true;
280             }
281             else
282             {
283                 Network<?, ?> n = getSubNetworkConsistNode(deleteThis);
284                 n.remove(deleteThis);
285                 return true;
286             }
287         }
288         else
289         {
290             throw new NetworkException("Deleting" + deleteThis.getId().toString() + "is failed. Possible cause:"
291                     + " node is not a member of the given Network");
292         }
293     }
