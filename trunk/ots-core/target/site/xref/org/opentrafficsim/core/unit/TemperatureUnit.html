19  public class TemperatureUnit extends OffsetUnit<TemperatureUnit>
20  {
21      /** */
22      private static final long serialVersionUID = 20140605L;
23  
24      /** Kelvin. */
25      public static final TemperatureUnit KELVIN;
26  
27      /** Degree Celsius. */
28      public static final TemperatureUnit DEGREE_CELSIUS;
29  
30      /** Degree Fahrenheit. */
31      public static final TemperatureUnit DEGREE_FAHRENHEIT;
32  
33      /** Degree Rankine. */
34      public static final TemperatureUnit DEGREE_RANKINE;
35  
36      /** Degree Reaumur. */
37      public static final TemperatureUnit DEGREE_REAUMUR;
38  
39      static
40      {
41          KELVIN = new TemperatureUnit("TemperatureUnit.kelvin", "TemperatureUnit.K", SI_BASE, 1.0, 0.0);
42          DEGREE_CELSIUS =
43                  new TemperatureUnit("TemperatureUnit.degree_Celsius", "TemperatureUnit.dgC", SI_DERIVED, 1.0, -273.15);
44          DEGREE_FAHRENHEIT =
45                  new TemperatureUnit("TemperatureUnit.degree_Fahrenheit", "TemperatureUnit.dgF", IMPERIAL, 5.0 / 9.0,
46                          -459.67);
47          DEGREE_RANKINE =
48                  new TemperatureUnit("TemperatureUnit.degree_Rankine", "TemperatureUnit.dgR", OTHER, 5.0 / 9.0, 0.0);
49          DEGREE_REAUMUR =
50                  new TemperatureUnit("TemperatureUnit.degree_Reaumur", "TemperatureUnit.dgRe", OTHER, 4.0 / 5.0, -273.15);
51      }
52  
53      /**
54       * @param nameKey the key to the locale file for the long name of the unit
55       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
56       * @param unitSystem the unit system, e.g. SI or Imperial
57       * @param conversionFactorToStandardUnit multiply by this number to convert to the standard unit
58       * @param offsetToKelvin the offsetToKelvin to add to convert to the standard (e.g., SI) unit
59       */
60      public TemperatureUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
61              final double conversionFactorToStandardUnit, final double offsetToKelvin)
62      {
63          super(nameKey, abbreviationKey, unitSystem, KELVIN, conversionFactorToStandardUnit, offsetToKelvin);
64      }
65  
66      /**
67       * @param nameKey the key to the locale file for the long name of the unit
68       * @param abbreviationKey the key to the locale file for the abbreviation of the unit
69       * @param unitSystem the unit system, e.g. SI or Imperial
70       * @param referenceUnit the unit to convert to
71       * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
72       *            reference unit
73       * @param offsetToKelvin the offsetToKelvin to add to convert to the standard (e.g., SI) unit
74       */
75      public TemperatureUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
76              final TemperatureUnit referenceUnit, final double conversionFactorToReferenceUnit,
77              final double offsetToKelvin)
78      {
79          super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, offsetToKelvin);
80      }
81  
82      /** {@inheritDoc} */
83      @Override
84      public final TemperatureUnit getStandardUnit()
85      {
86          return KELVIN;
87      }
88  
89      /** {@inheritDoc} */
90      @Override
91      public final String getSICoefficientsString()
92      {
93          return "K";
94      }
95  
96  }
