154     /**
155      * @param massUnit the unit of mass for the energy unit, e.g., kilogram
156      * @param lengthUnit the unit of length for the energy unit, e.g., meter
157      * @param timeUnit the unit of time for the energy unit, e.g., second
158      * @param nameKey the key to the locale file for the long name of the unit
159      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
160      * @param unitSystem the unit system, e.g. SI or Imperial
161      */
162     public EnergyUnit(final MassUnit massUnit, final LengthUnit lengthUnit, final TimeUnit timeUnit,
163             final String nameKey, final String abbreviationKey, final UnitSystem unitSystem)
164     {
165         super(nameKey, abbreviationKey, unitSystem, JOULE, massUnit.getConversionFactorToStandardUnit()
166                 * lengthUnit.getConversionFactorToStandardUnit() * lengthUnit.getConversionFactorToStandardUnit()
167                 / (timeUnit.getConversionFactorToStandardUnit() * timeUnit.getConversionFactorToStandardUnit()), true);
168         this.massUnit = massUnit;
169         this.lengthUnit = lengthUnit;
170         this.timeUnit = timeUnit;
171     }
172 
173     /**
174      * @param lengthUnit the unit of length for the energy unit, e.g., m
175      * @param forceUnit the unit of force for the energy unit, e.g., Newton
176      * @param nameKey the key to the locale file for the long name of the unit
177      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
178      * @param unitSystem the unit system, e.g. SI or Imperial
179      */
180     public EnergyUnit(final LengthUnit lengthUnit, final ForceUnit forceUnit, final String nameKey,
181             final String abbreviationKey, final UnitSystem unitSystem)
182     {
183         super(nameKey, abbreviationKey, unitSystem, JOULE, forceUnit.getConversionFactorToStandardUnit()
184                 * lengthUnit.getConversionFactorToStandardUnit(), true);
185         this.massUnit = forceUnit.getMassUnit();
186         this.lengthUnit = forceUnit.getLengthUnit();
187         this.timeUnit = forceUnit.getTimeUnit();
188     }
189 
190     /**
191      * @param nameKey the key to the locale file for the long name of the unit
192      * @param abbreviationKey the key to the locale file for the abbreviation of the unit
193      * @param unitSystem the unit system, e.g. SI or Imperial
194      * @param referenceUnit the unit to convert to
195      * @param conversionFactorToReferenceUnit multiply a value in this unit by the factor to convert to the given
196      *            reference unit
197      */
198     public EnergyUnit(final String nameKey, final String abbreviationKey, final UnitSystem unitSystem,
199             final EnergyUnit referenceUnit, final double conversionFactorToReferenceUnit)
200     {
201         super(nameKey, abbreviationKey, unitSystem, referenceUnit, conversionFactorToReferenceUnit, true);
202         this.massUnit = referenceUnit.getMassUnit();
203         this.lengthUnit = referenceUnit.getLengthUnit();
204         this.timeUnit = referenceUnit.getTimeUnit();
205     }
