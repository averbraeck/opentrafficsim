22  public final class SafeLaneChange

24      /**
25       * This class should never be instantiated.
26       */
27      private SafeLaneChange()
28      {
29          // This class should never be instantiated.
30      }
31  
32      /**
33       * Determine if dangerous decelerations are incurred if a Car changes into a lane where another Car is driving. <br>
34       * This implements the <i>Safety Criterion</i> as described in Traffic Flow Dynamics by Martin Treiber and Arne
35       * Kesting, ISBN 978-3-642-32459-8 ISBN 978-3-642-32460-4 (eBook), 2013, Chapter 14.3.1, pp 242-243.
36       * @param referenceCar Car; the car following model of this car is used to determine the needed acceleration
37       *            (deceleration).
38       * @param otherCar Car; the car driving in the other lane
39       * @param maximumDeceleration DoubleScalar.Abs&lt;AccelerationUnit&gt;; the maximum (considered safe) deceleration
40       *            (must be positive; something on the order of 2m/s/s)
41       * @param speedLimit DoubleScalar.Abs&lt;SpeedUnit&gt;; the speed limit
42       * @return Boolean; true if the resulting deceleration is safe; false if the resulting deceleration is unsafe
43       */
44      public static boolean safe(final Car referenceCar, final Car otherCar,
45              final DoubleScalar.Rel<AccelerationUnit> maximumDeceleration, final DoubleScalar.Abs<SpeedUnit> speedLimit)
46      {
47          DoubleScalar.Abs<TimeUnit> when = referenceCar.getNextEvaluationTime();
48          CarFollowingModel carFollowingModel = referenceCar.getCarFollowingModel();
49          if (referenceCar.getPosition(when).getValueSI() > otherCar.getPosition(when).getValueSI())
50          { // The referenceCar is ahead of the otherCar
51              return FollowAcceleration.acceleration(otherCar, referenceCar, when, carFollowingModel, speedLimit)
52                      .getValueSI() >= -maximumDeceleration.getValueSI();
53          }
54          // The otherCar is exactly parallel or ahead of the referenceCar
55          return FollowAcceleration.acceleration(referenceCar, otherCar, when, carFollowingModel, speedLimit)
56                  .getValueSI() >= -maximumDeceleration.getValueSI();
57      }
58  }
