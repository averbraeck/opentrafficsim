3   import java.rmi.RemoteException;
4   import java.util.Collection;
5   
6   import org.opentrafficsim.car.Car;
7   import org.opentrafficsim.core.unit.AccelerationUnit;
8   import org.opentrafficsim.core.unit.SpeedUnit;
9   import org.opentrafficsim.core.unit.TimeUnit;

12  /**
13   * Abstract car following model.
14   * <p>
15   * Copyright (c) 2013-2014 Delft University of Technology, PO Box 5, 2600 AA, Delft, the Netherlands. All rights reserved. <br>
16   * BSD-style license. See <a href="http://opentrafficsim.org/node/13">OpenTrafficSim License</a>.
17   * <p>
18   * @version Jul 2, 2014 <br>
19   * @author <a href="http://www.tudelft.nl/pknoppers">Peter Knoppers</a>
20   * @param <C> Car;
21   */
22  public interface CarFollowingModel<C extends Car>
23  {
24      /**
25       * Compute the acceleration that would be used to follow a set of leaders.
26       * @param car the Car for which acceleration is computed
27       * @param leaders Set&lt;Car&gt;; the set of leaders to take into consideration
28       * @param speedLimit DoubleScalarAbs&lt;SpeedUnit&gt;; the local speed limit
29       * @return CarFollowingModelResult; the result of application of the car following model
30       */
31      CarFollowingModelResult computeAcceleration(final C car, final Collection<C> leaders,
32              final DoubleScalar.Abs<SpeedUnit> speedLimit);
33  
34      /**
35       * Compute the acceleration and lane change.
36       * @param car Car; the Car for which the acceleration and lane change is computed
37       * @param sameLaneCars Collection&lt;Car&gt;; the set of observable Cars in the current lane (can not be null)
38       * @param preferredLaneCars Collection&lt;Car&gt;; the set of observable Cars in the adjacent lane where cars should drive
39       *            in the absence of other traffic (must be null if there is no such lane)
40       * @param nonPreferredLaneCars Collection&lt;Car&gt;; the set of observable Cars in the adjacent lane into which cars should
41       *            merge to overtake other traffic (must be null if there is no such lane)
42       * @param speedLimit DoubleScalarAbs&lt;SpeedUnit&gt;; the local speed limit
43       * @param preferredLaneRouteIncentive Double; route incentive to merge to the adjacent lane where cars should drive in the
44       *            absence of other traffic
45       * @param nonPreferredLaneRouteIncentive Double; route incentive to merge to the adjacent lane into which cars should merge
46       *            to overtake other traffic
47       * @return CarFollowingModelResult; the result of the lane change and car following model
48       * @throws RemoteException
49       */
50      CarFollowingModelResult computeLaneChangeAndAcceleration(final C car, final Collection<C> sameLaneCars,
51              final Collection<C> preferredLaneCars, final Collection<C> nonPreferredLaneCars,
52              final DoubleScalar.Abs<SpeedUnit> speedLimit, double preferredLaneRouteIncentive,
53              double nonPreferredLaneRouteIncentive) throws RemoteException;
54

83      class CarFollowingModelResult
84      {
85          /** Acceleration that will be maintained during the current time slot. */
86          public final DoubleScalar.Abs<AccelerationUnit> acceleration;
87  
88          /** Time when the current time slot ends. */
89          public final DoubleScalar.Abs<TimeUnit> validUntil;
90  
91          /**
92           * Lane change; 0: stay in current lane; -1 merge onto adjacent overtaking lane; +1 merge towards the default lane.
93           */
94          public final int laneChange;
95  
96          /**
97           * Create a new CarFollowingModelResult.
98           * @param acceleration DoubleScalarAbs&lt;AccelerationUnit&gt;; computed acceleration
99           * @param validUntil DoubleScalarAbs&lt;TimeUnit&gt;; time when this result expires
100          * @param laneChange Integer; the lane determined change; 0: stay in current lane; -1 merge onto adjacent overtaking
101          *            lane; +1 merge towards the default lane
102          */
103         public CarFollowingModelResult(final DoubleScalar.Abs<AccelerationUnit> acceleration,
104                 final DoubleScalar.Abs<TimeUnit> validUntil, final int laneChange)
105         {
106             this.acceleration = acceleration;
107             this.validUntil = validUntil;
108             this.laneChange = laneChange;
109         }
110     }
