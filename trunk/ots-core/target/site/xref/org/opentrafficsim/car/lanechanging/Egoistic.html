24  public final class Egoistic
25  {
26      /**
27       * This class should never be instantiated.
28       */
29      private Egoistic()
30      {
31          // This class should never be instantiated.
32      }
33  
34      /**
35       * Compute the acceleration of this car and the new follower car after a considered lane change. The lowers of the
36       * two computed accelerations is returned. If changing lane is not possible because it would result in dangerous
37       * deceleration or collision, the returned value is Double.NEGATIVE_INFINITY.
38       * @param referenceCar Car; the car that considers changing lane
39       * @param otherCars Set&lt;Car&gt;; the cars in the target lane
40       * @param maximumDeceleration DoubleScalar.Abs&lt;AccelerationUnit&gt;; the maximum (deemed safe) deceleration. This
41       *            must be a positive value, about 2 m/s/s.
42       * @param speedLimit DoubleScalar.Abs&lt;SpeedUnit&gt;; the speed limit
43       * @return DoubleScalar.Abs&lt;AccelerationUnit&gt;; the lowest acceleration (highest deceleration) incurred if the
44       *         lane change is carried out
45       */
46      public static DoubleScalar.Abs<AccelerationUnit> acceleration(final Car referenceCar, final Set<Car> otherCars,
47              final DoubleScalar.Rel<AccelerationUnit> maximumDeceleration, final DoubleScalar.Abs<SpeedUnit> speedLimit)
48      {
49          Car leader = null;
50          DoubleScalar.Rel<LengthUnit> leaderHeadway = null;
51          Car follower = null;
52          DoubleScalar.Rel<LengthUnit> followerHeadway = null;
53          DoubleScalar.Abs<TimeUnit> when = referenceCar.getNextEvaluationTime();
54          DoubleScalar.Abs<LengthUnit> referenceCarPosition = referenceCar.getPosition(when);
55          // TODO: if otherCars are sorted in some way, scanning the entire list like this is not needed.
56  
57          // TODO: the car following model already deals with a Set of leaders; can't we do something similar with the set
58          // of followers?
59          for (Car c : otherCars)
60          {
61              DoubleScalar.Rel<LengthUnit> headway =
62                      MutableDoubleScalar.minus(c.getPosition(when), referenceCarPosition).immutable();
63              if (headway.getValueSI() < 0)
64              {
65                  if (null == follower || followerHeadway.getValueSI() > headway.getValueSI())
66                  {
67                      follower = c;
68                      followerHeadway = headway;
69                  }
70              }
71              else
72              {
73                  if (null == leader || leaderHeadway.getValueSI() < headway.getValueSI())
74                  {
75                      leader = c;
76                      leaderHeadway = headway;
77                  }
78              }
79          }
80          CarFollowingModel carFollowingModel = referenceCar.getCarFollowingModel();
81          DoubleScalar.Abs<AccelerationUnit> followerAcceleration =
82                  FollowAcceleration.acceleration(follower, referenceCar, when, carFollowingModel, speedLimit);
83          if (followerAcceleration.getValueSI() < -maximumDeceleration.getValueSI())
84          {
85              return new DoubleScalar.Abs<AccelerationUnit>(Double.NEGATIVE_INFINITY, AccelerationUnit.METER_PER_SECOND_2);
86          }
87          DoubleScalar.Abs<AccelerationUnit> referenceAcceleration =
88                  FollowAcceleration.acceleration(referenceCar, leader, when, carFollowingModel, speedLimit);
89          if (followerAcceleration.getValueSI() < referenceAcceleration.getValueSI())
90          {
91              return followerAcceleration;
92          }
93          return referenceAcceleration;
94      }
