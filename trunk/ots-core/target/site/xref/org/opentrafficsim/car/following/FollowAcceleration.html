14  /**
15   * Determine acceleration (deceleration) for a Car that follows another car.
16   * <p>

21   * @author <a href="http://www.tudelft.nl/pknoppers">Peter Knoppers</a>
22   */
23  public final class FollowAcceleration
24  {
25      /**
26       * This class should never be instantiated.
27       */
28      private FollowAcceleration()
29      {
30          // This class should never be instantiated
31      }
32  
33      /**
34       * Compute the acceleration (or deceleration) for a Car following another Car.
35       * @param follower Car; the car that is following
36       * @param leader Car; the car that is leading
37       * @param when DoubleScalar.Abs&lt;TimeUnit&gt;; the current time
38       * @param carFollowingModel CarFollowingModel; the car following model that is used to compute the result
39       * @param speedLimit DoubleScalar.Abs&lt;SpeedUnit&gt;; the speed limit
40       * @return DoubleScalar.Abs&lt;AccelerationUnit&gt;; the acceleration (deceleration) for the following car in order to not
41       *         collide with the leader car
42       */
43      public static DoubleScalar.Abs<AccelerationUnit> acceleration(final Car follower, final Car leader,
44              final DoubleScalar.Abs<TimeUnit> when, final CarFollowingModel carFollowingModel,
45              final DoubleScalar.Abs<SpeedUnit> speedLimit)
46      {
47          if (null != leader)
48          {
49              MutableDoubleScalar.Rel<LengthUnit> headway =
50                      MutableDoubleScalar.minus(leader.positionOfRear(when), follower.positionOfFront(when));
51              if (headway.getValueSI() <= 0)
52              { // Immediate collision; return a prohibitive negative value
53                  return new DoubleScalar.Abs<AccelerationUnit>(Double.NEGATIVE_INFINITY, AccelerationUnit.METER_PER_SECOND_2);
54              }
55          }
56          // Wrap the leader in a set, then apply the car following model
57          Set<Car> leaders = new HashSet<Car>(1);
58          if (null != leader)
59          {
60              leaders.add(leader);
61          }
62          return carFollowingModel.computeAcceleration(follower, leaders, speedLimit).acceleration;
63      }

65  }
