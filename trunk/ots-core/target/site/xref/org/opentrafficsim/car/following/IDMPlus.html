16  /**
17   * IDMPlus implements the <i>Integrated Lane Change Model with Relaxation and Synchronization</i> as published by Wouter J.
18   * Schakel, Victor L. Knoop and Bart van Arem in Transportation Research Record No 2316 pp 47-57, Washington D.C., 2012. <br>
19   * There are two nasty type setting errors in equation 7 in this published version of the paper. Both times an equals sign
20   * (<cite>=</cite>) after <cite>a<sub>gain</sub></cite> should <b>not</b> be there.
21   * <p>
22   * Copyright (c) 2013-2014 Delft University of Technology, PO Box 5, 2600 AA, Delft, the Netherlands. All rights reserved. <br>
23   * BSD-style license. See <a href="http://opentrafficsim.org/node/13">OpenTrafficSim License</a>.
24   * <p>
25   * @version Jul 4, 2014 <br>
26   * @author <a href="http://www.tudelft.nl/pknoppers">Peter Knoppers</a>
27   * @param <Line>
28   * @param <C> Car; 
29   */
30  public class IDMPlus<Line, C extends Car> implements CarFollowingModel<C>
31  {
32      /** Preferred net longitudinal distance when stopped [m]. */
33      protected final DoubleScalar.Rel<LengthUnit> s0 = new DoubleScalar.Rel<LengthUnit>(3, LengthUnit.METER);
34  
35      /** Longitudinal acceleration [m/s^2]. */
36      protected final DoubleScalar.Abs<AccelerationUnit> a = new DoubleScalar.Abs<AccelerationUnit>(1.25,
37              AccelerationUnit.METER_PER_SECOND_2);
38  
39      /** Regular longitudinal acceleration [m/s^2]. */
40      protected final DoubleScalar.Abs<AccelerationUnit> aMin = new DoubleScalar.Abs<AccelerationUnit>(1.25,
41              AccelerationUnit.METER_PER_SECOND_2);
42  
43      /** Longitudinal deceleration [m/s^2]. (Should be a positive value even though it is a <b>de</b>celeration.) */
44      protected final DoubleScalar.Abs<AccelerationUnit> b = new DoubleScalar.Abs<AccelerationUnit>(2.09,
45              AccelerationUnit.METER_PER_SECOND_2);
46  
47      /**
48       * Maximum deceleration when actual speed is more than desired speed (v &gt; vdes). (Should be a positive value even though
49       * it is a <b>de</b>celeration.) [m/s^2]
50       */
51      protected final DoubleScalar.Abs<AccelerationUnit> b0 = new DoubleScalar.Abs<AccelerationUnit>(0.5,
52              AccelerationUnit.METER_PER_SECOND_2);
53  
54      /** Longitudinal regular following headway [s]. */
55      protected final DoubleScalar.Rel<TimeUnit> tMax = new DoubleScalar.Rel<TimeUnit>(1.2, TimeUnit.SECOND);
56  
57      /** LMRS free lane change threshold. (Value between 0 and dSync) */
58      protected final double dFree = .365;
59  
60      /** LMRS synchronized lane change threshold. (Value between dFree and dCoop) */
61      protected final double dSync = .577;
62  
63      /** LMRS cooperative lane change threshold. (Value between dSync and 1.0) */
64      protected final double dCoop = .788;
65  
66      /** LMRS mandatory lane change time [s]. */
67      protected final DoubleScalar.Rel<TimeUnit> t0 = new DoubleScalar.Rel<TimeUnit>(43, TimeUnit.SECOND);
68  
69      /** LMRS mandatory lane change distance [m]. */
70      protected final DoubleScalar.Rel<LengthUnit> x0 = new DoubleScalar.Rel<LengthUnit>(295, LengthUnit.METER);
71  
72      /** LMRS speed gain [m/s] for full desire. (The paper specifies this value in [km/h]) */
73      protected final DoubleScalar.Abs<SpeedUnit> vGain = new DoubleScalar.Abs<SpeedUnit>(69.6, SpeedUnit.KM_PER_HOUR);
74  
75      /** LMRS critical speed [m/s] for a speed gain in the right lane. (The paper specifies this value in [km/h]) */
76      protected final DoubleScalar.Abs<SpeedUnit> vCong = new DoubleScalar.Abs<SpeedUnit>(60, SpeedUnit.KM_PER_HOUR);
77  
78      /** Safe time headway. */
79      protected final DoubleScalar.Rel<TimeUnit> tSafe = new DoubleScalar.Rel<TimeUnit>(1.6, TimeUnit.SECOND);
80  
81      /**
82       * Mean speed limit adherence (1.0: mean free speed equals the speed limit; 1.1: mean speed limit equals 110% of the speed
83       * limit, etc.).
84       */
85      protected final double delta = 1.0;
86  
87      /**
88       * Time slot size used by IDMPlus (not defined in the paper, but 0.5s is a reasonable trade-off between computational speed
89       * and accuracy).
90       */
91      protected final DoubleScalar.Rel<TimeUnit> stepSize = new DoubleScalar.Rel<TimeUnit>(0.5, TimeUnit.SECOND);
92  
93      /**
94       * Desired speed (taking into account the urge to drive a little faster or slower than the posted speed limit).
95       * @param car Car; the Car whose desired speed must be returned
96       * @param speedLimit DoubleScalarAbs&lt;SpeedUnit&gt;; the speed limit
97       * @return DoubleScalarRel&lt;SpeedUnit&gt;; the desired speed
98       */
99      private DoubleScalar.Rel<SpeedUnit> vDes(final C car, final DoubleScalar.Abs<SpeedUnit> speedLimit)
100     {
101         return new DoubleScalar.Rel<SpeedUnit>(Math.min(this.delta * speedLimit.getValueSI(), car.vMax().getValueSI()),
102                 SpeedUnit.METER_PER_SECOND);
103     }

105     /** {@inheritDoc} */
106     @Override
107     public final CarFollowingModelResult computeAcceleration(final C car, final Collection<C> leaders,
108             final DoubleScalar.Abs<SpeedUnit> speedLimit)
109     {
110         DoubleScalar.Abs<TimeUnit> thisEvaluationTime = car.getNextEvaluationTime();
111         // System.out.println("evaluation time is " + thisEvaluationTime);
112         // System.out.println("vDes is " + vDes);
113         DoubleScalar.Abs<LengthUnit> myFrontPosition = car.positionOfFront(thisEvaluationTime);
114         // System.out.println("myFrontPosition is " + myFrontPosition);
115         DoubleScalar.Rel<LengthUnit> shortestHeadway = new DoubleScalar.Rel<LengthUnit>(Double.MAX_VALUE, LengthUnit.METER);
116         C closestLeader = null;
117         for (C leader : leaders)
118         {
119             DoubleScalar.Rel<LengthUnit> s =
120                     MutableDoubleScalar.minus(leader.positionOfRear(thisEvaluationTime), myFrontPosition).immutable();
121             // System.out.println("s is " + s);
122             if (s.getValueSI() < 0)
123             {
124                 continue; // Ignore cars that are behind this car
125             }
126             if (s.getValueSI() < shortestHeadway.getValueSI())
127             {
128                 shortestHeadway = s;
129                 closestLeader = leader;
130             }
131         }
132         // System.out.println("shortestHeadway is " + shortestHeadway);
133         DoubleScalar.Rel<SpeedUnit> myCurrentSpeed = car.getVelocity(thisEvaluationTime);
134         double speedIncentive = 1 - Math.pow(myCurrentSpeed.getValueSI() / vDes(car, speedLimit).getValueSI(), 4);
135         // System.out.println("speedIncentive is " + speedIncentive);
136         MutableDoubleScalar.Rel<AccelerationUnit> logWeightedAverageSpeedTimes2 =
137                 new MutableDoubleScalar.Rel<AccelerationUnit>(Math.sqrt(this.a.getValueSI() * this.b.getValueSI()),
138                         AccelerationUnit.METER_PER_SECOND_2);
139         logWeightedAverageSpeedTimes2.multiply(2); // don't forget the times 2
140         DoubleScalar.Rel<SpeedUnit> dV =
141                 (null == closestLeader) ? new DoubleScalar.Rel<SpeedUnit>(0, SpeedUnit.METER_PER_SECOND) : MutableDoubleScalar
142                         .minus(car.getVelocity(thisEvaluationTime), closestLeader.getVelocity(thisEvaluationTime)).immutable();
143         // System.out.println("dV is " + dV);
144         // System.out.println(" v is " + car.speed(thisEvaluationTime));
145         // System.out.println("s0 is " + this.s0);
146         DoubleScalar.Rel<LengthUnit> sStar =
147                 MutableDoubleScalar.plus(
148                         LengthUnit.METER,
149                         this.s0,
150                         Calc.speedTimesTime(car.getVelocity(thisEvaluationTime), this.tSafe),
151                         Calc.speedTimesTime(dV,
152                                 Calc.speedDividedByAcceleration(myCurrentSpeed, logWeightedAverageSpeedTimes2.immutable())))
153                         .immutable();
154         if (sStar.getValueSI() < 0) // Negative value should be treated as 0
155         {
156             sStar = new DoubleScalar.Rel<LengthUnit>(0, LengthUnit.METER);
157         }
158         // System.out.println("s* is " + sStar);
159         double distanceIncentive = 1 - Math.pow(sStar.getValueSI() / shortestHeadway.getValueSI(), 2);
160         MutableDoubleScalar.Abs<AccelerationUnit> newAcceleration = new MutableDoubleScalar.Abs<AccelerationUnit>(this.a);
161         newAcceleration.multiply(Math.min(speedIncentive, distanceIncentive));
162         // System.out.println("distanceIncentive is " + distanceIncentive);
163         // System.out.println("newAcceleration is " + newAcceleration);
164         MutableDoubleScalar.Abs<TimeUnit> nextEvaluationTime = thisEvaluationTime.mutable();
165         nextEvaluationTime.add(this.stepSize);
166         return new CarFollowingModelResult(newAcceleration.immutable(), nextEvaluationTime.immutable(), 0);
167     }

171     @Override
172     public final CarFollowingModelResult computeLaneChangeAndAcceleration(final C car, final Collection<C> sameLaneCars,
173             final Collection<C> preferredLaneCars, final Collection<C> nonPreferredLaneCars,
174             final DoubleScalar.Abs<SpeedUnit> speedLimit, final double preferredLaneRouteIncentive,
175             final double nonPreferredLaneRouteIncentive) throws RemoteException
176     {
177         System.out.println(String.format(
178                 "Route desire to merge to preferredLane: %.3f, route desire to merge to overtakingLane: %.3f",
179                 preferredLaneRouteIncentive, nonPreferredLaneRouteIncentive));
180         DoubleScalar.Abs<SpeedUnit> vAntStraight = anticipatedSpeed(speedLimit, car, sameLaneCars);
181         double aGain =
182                 (this.a.getValueSI() - Math
183                         .max(computeAcceleration(car, sameLaneCars, speedLimit).acceleration.getValueSI(), 0))
184                         / this.a.getValueSI();
185         System.out.println(String.format("aGain: %.3f", aGain));
186         double nonPreferredLaneSpeedIncentive = 0;
187         if (null != nonPreferredLaneCars)
188         {
189             nonPreferredLaneSpeedIncentive =
190                     aGain
191                             * MutableDoubleScalar.minus(anticipatedSpeed(speedLimit, car, nonPreferredLaneCars), vAntStraight)
192                                     .getValueSI() / this.vGain.getValueSI();
193         }
194         double dBias = 0;
195         double preferredLaneSpeedIncentive = 0;
196         if (null != preferredLaneCars)
197         {
198             DoubleScalar.Abs<SpeedUnit> vAnt = anticipatedSpeed(speedLimit, car, preferredLaneCars);
199             if (vAnt.getValueSI() > this.vCong.getValueSI())
200             {
201                 preferredLaneSpeedIncentive =
202                         MutableDoubleScalar.minus(vAnt, vAntStraight).getValueSI() / this.vGain.getValueSI();
203                 if (preferredLaneSpeedIncentive > 0)
204                 {
205                     preferredLaneSpeedIncentive = 0; // changing lane to overtake "on the right" is not permitted
206                 }
207             }
208             else
209             {
210                 preferredLaneSpeedIncentive =
211                         aGain * MutableDoubleScalar.minus(vAnt, vAntStraight).getValueSI() / this.vGain.getValueSI();
212             }
213             // FIXME: comparing double values for equality is not "reliable"
214             if (preferredLaneRouteIncentive >= 0 && vAnt.getValueSI() == vDes(car, speedLimit).getValueSI())
215             {
216                 dBias = this.dFree;
217             }
218         }
219         System.out.println(String.format(
220                 "Speed desire to merge to preferredLane: %.3f, speed desire to merge to overtakingLane: %.3f",
221                 preferredLaneSpeedIncentive, nonPreferredLaneSpeedIncentive));
222         double preferredLaneChangeDesire = totalDesire(preferredLaneRouteIncentive, preferredLaneSpeedIncentive, dBias);
223         double nonPreferredLaneChangeDesire = totalDesire(nonPreferredLaneRouteIncentive, nonPreferredLaneSpeedIncentive, 0);
224         boolean preferredLaneChangeOK = checkLaneChange(car, preferredLaneCars, preferredLaneChangeDesire, speedLimit);
225         CarFollowingModelResult straight = computeAcceleration(car, sameLaneCars, speedLimit);
226         System.out.println(String.format(
227                 "Total desire to merge to preferredLane: %.3f, total desire to merge to overtakingLane: %.3f",
228                 preferredLaneChangeDesire, nonPreferredLaneChangeDesire));
229         return null;
230     }

232     /**
233      * Check if a lane change can be executed safely.
234      * @param car Car; the Car that considers changing lane
235      * @param carsInOtherLane Collection&lt;Car&gt; the car in the adjacent lane
236      * @param desire double; the desire to change into the adjacent lane
237      * @param speedLimit DoubleScalarAbs&lt;SpeedUnit&gt;; the speed limit in the adjacent lane
238      * @return boolean; true if the lane change can be performed; false if the lane change should not be performed
239      * @throws RemoteException 
240      */
241     private boolean checkLaneChange(final C car, final Collection<C> carsInOtherLane, final double desire,
242             final DoubleScalar.Abs<SpeedUnit> speedLimit) throws RemoteException
243     {
244         // Find the new leader and follower
245         C leader = null;
246         DoubleScalar.Rel<LengthUnit> leaderHeadway = null;
247         C follower = null;
248         DoubleScalar.Rel<LengthUnit> followerHeadway = null;
249         for (C c : carsInOtherLane)
250         {
251             DoubleScalar.Rel<LengthUnit> headway = car.headway(c);
252             if (headway.getValueSI() > 0)
253             {
254                 if (null == leader || headway.getValueSI() < leaderHeadway.getValueSI())
255                 {
256                     leader = c;
257                     leaderHeadway = headway;
258                 }
259             }
260             else
261             {
262                 if (null == follower || headway.getValueSI() > followerHeadway.getValueSI())
263                 {
264                     follower = c;
265                     followerHeadway = headway;
266                 }
267             }
268         }
269         Collection<C> leaders = new ArrayList<C>();
270         if (null != leader)
271         {
272             leaders.add(leader);
273         }
274         DoubleScalar.Abs<AccelerationUnit> carAcceleration = computeAcceleration(car, leaders, speedLimit).acceleration;
275         if (carAcceleration.getValueSI() < this.b.getValueSI())
276         {
277             return false; // leader would be too close
278         }
279         if (null != follower)
280         {
281             Collection<C> referenceCarGroup = new ArrayList<C>();
282             referenceCarGroup.add(car);
283             DoubleScalar.Abs<AccelerationUnit> otherCarAcceleration =
284                     computeAcceleration(follower, referenceCarGroup, speedLimit).acceleration;
285             // This assumes that the follower also uses IDMPlus (which is not unreasonable)
286             if (otherCarAcceleration.getValueSI() < this.b.getValueSI())
287             {
288                 return false; // follower would be too close
289             }
290         }
291         return true;
292     }

294     /**
295      * Compute the total desire to change lane. <br>
296      * Equation 1 in the LMRS paper.
297      * @param routeIncentive double; the desire to make the lane change in order to follow the route
298      * @param speedIncentive double; the desire to make the lane change in order to gain speed (or avoid reducing speed)
299      * @param bias double; the desire to make the lane change in order to get to into the preferred lane
300      * @return double; the total desire to change lane
301      */
302     private double totalDesire(final double routeIncentive, final double speedIncentive, final double bias)
303     {
304         return routeIncentive + theta(routeIncentive, speedIncentive) * (speedIncentive + bias);
305     }
306 
307     /**
308      * Combine the lane change incentives into one value that describes how <i>voluntary</i> a potential lane change is. <br>
309      * Equation 11 in the LRMS paper.
310      * @param routeIncentive double; the lane change incentive to follow the intended route
311      * @param speedIncentive double; the lane change incentive for gaining (or not so much reducing) speed
312      * @return double; a value between 0 (no urge to change lane) and 1 (maximum urge to change lane)
313      */
314     private double theta(final double routeIncentive, final double speedIncentive)
315     {
316         if (routeIncentive * speedIncentive < 0 && Math.abs(routeIncentive) >= this.dCoop)
317         {
318             return 0;
319         }
320         else if (routeIncentive * speedIncentive >= 0 || Math.abs(routeIncentive) <= this.dSync)
321         {
322             return 1;
323         }
324         else
325         {
326             return (this.dCoop - Math.abs(routeIncentive)) / (this.dCoop - this.dSync);
327         }
328     }

330     /**
331      * Compute the anticipated speed if driving in a lane among a set of other cars.
332      * @param speedLimit DoubleScalarAbs&lt;SpeedUnit&gt;; the speedLimit in the lane
333      * @param car Car; the reference car (only the current position of the reference car is used)
334      * @param leaders Collection&lt;Car&gt;; the set of other cars
335      * @return DoubleScalarAbs&lt;SpeedUnit&gt;; the anticipated speed
336      */
337     private DoubleScalar.Abs<SpeedUnit> anticipatedSpeed(final DoubleScalar.Abs<SpeedUnit> speedLimit, final C car,
338             final Collection<C> leaders)
339     {
340         DoubleScalar.Abs<SpeedUnit> result = speedLimit;
341         DoubleScalar.Abs<LengthUnit> frontPositionOfCar = car.positionOfFront(car.getNextEvaluationTime());
342         for (C leader : leaders)
343         {
344             DoubleScalar.Rel<LengthUnit> headway =
345                     MutableDoubleScalar.minus(leader.positionOfRear(car.getNextEvaluationTime()), frontPositionOfCar)
346                             .immutable();
347             if (headway.getValueSI() < 0)
348             {
349                 continue;
350             }
351             if (headway.getValueSI() > this.x0.getValueSI())
352             {
353                 continue;
354             }
355             DoubleScalar.Rel<SpeedUnit> leaderSpeed = leader.getVelocity(car.getNextEvaluationTime());
356             if (leaderSpeed.getValueSI() < result.getValueSI())
357             {
358                 result = new DoubleScalar.Abs<SpeedUnit>(leaderSpeed.getValueSI(), leaderSpeed.getUnit());
359             }
360         }
361         return result;
362     }
