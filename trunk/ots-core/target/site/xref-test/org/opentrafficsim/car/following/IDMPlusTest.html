30      /**
31       * Test IDMPlus.
32       */
33      @SuppressWarnings("static-method")
34      @Test
35      public final void computeAcceleration()
36      {
37          // Check a car standing still with no leaders accelerates with maximum acceleration
38          OTSDEVSSimulator simulator = new OTSDEVSSimulator();
39          CarFollowingModel<Car> carFollowingModel = new IDMPlus<Line<String>, Car>();
40          DoubleScalar.Abs<TimeUnit> initialTime = new DoubleScalar.Abs<TimeUnit>(0, TimeUnit.SECOND);
41          DoubleScalar.Abs<LengthUnit> initialPosition = new DoubleScalar.Abs<LengthUnit>(123.456, LengthUnit.METER);
42          DoubleScalar.Rel<SpeedUnit> initialSpeed = new DoubleScalar.Rel<SpeedUnit>(0, SpeedUnit.KM_PER_HOUR);
43          Car referenceCar = new Car(12345, simulator, carFollowingModel, initialTime, initialPosition, initialSpeed);
44          DoubleScalar.Abs<SpeedUnit> speedLimit = new DoubleScalar.Abs<SpeedUnit>(100, SpeedUnit.KM_PER_HOUR);
45          Collection<Car> leaders = new ArrayList<Car>();
46          CarFollowingModelResult cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
47          assertEquals("Standard time slice in IDM+ is 0.5s", 0.5, cfmr.validUntil.getValueSI(), 0.0001);
48          assertEquals("Acceleration should be maximum", 1.25, cfmr.acceleration.getValueSI(), 0.0001);
49          // Create another car at exactly the stationary following distance
50          // Check that the follower remains stationary
51          DoubleScalar.Abs<LengthUnit> leaderPosition =
52                  new DoubleScalar.Abs<LengthUnit>(3 + referenceCar.length().getValueSI()
53                          + referenceCar.getPosition(initialTime).getValueSI(), LengthUnit.METER);
54          Car leaderCar = new Car(23456, simulator, null, initialTime, leaderPosition, initialSpeed);
55          leaders.add(leaderCar);
56          cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
57          assertEquals("Acceleration should be 0", 0, cfmr.acceleration.getValueSI(), 0.0001);
58          leaders.clear();
59          leaderPosition =
60                  new DoubleScalar.Abs<LengthUnit>(1000 + (3 + referenceCar.length().getValueSI() + referenceCar.getPosition(
61                          initialTime).getValueSI()), LengthUnit.METER);
62          // Exercise the if statement that ignores leaders that are further ahead
63          Car leaderCar2 = new Car(34567, simulator, null, initialTime, leaderPosition, initialSpeed);
64          leaders.add(leaderCar2); // Put the 2nd leader in first place
65          leaders.add(leaderCar);
66          cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
67          assertEquals("Acceleration should be 0", 0, cfmr.acceleration.getValueSI(), 0.0001);
68          leaders.clear();
69          leaders.add(leaderCar); // Put the 1st leader in first place
70          leaders.add(leaderCar2);
71          cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
72          assertEquals("Acceleration should be 0", 0, cfmr.acceleration.getValueSI(), 0.0001);
73          leaders.clear();
74          leaderPosition =
75                  new DoubleScalar.Abs<LengthUnit>(-(3 + referenceCar.length().getValueSI() + referenceCar.getPosition(
76                          initialTime).getValueSI()), LengthUnit.METER);
77          leaderCar = new Car(23456, simulator, null, initialTime, leaderPosition, initialSpeed);
78          leaders.add(leaderCar);
79          cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
80          assertEquals("Acceleration should be 1.25", 1.25, cfmr.acceleration.getValueSI(), 0.0001);
81          // Check that the returned acceleration increases with the distance to the leader
82          double referenceAcceleration = -1;
83          for (int spareDistance = 0; spareDistance <= 500; spareDistance++)
84          {
85              leaders.clear();
86              leaderPosition =
87                      new DoubleScalar.Abs<LengthUnit>(spareDistance
88                              + (3 + referenceCar.length().getValueSI() + referenceCar.getPosition(initialTime).getValueSI()),
89                              LengthUnit.METER);
90              leaderCar = new Car(0, simulator, null, initialTime, leaderPosition, initialSpeed);
91              leaders.add(leaderCar);
92              cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
93              double acceleration = cfmr.acceleration.getValueSI();
94              // System.out.println("Acceleration with stationary leader at " + spareDistance + " is " + acceleration);
95              assertTrue("acceleration should not decrease when distance to leader is increased",
96                      acceleration >= referenceAcceleration);
97              referenceAcceleration = acceleration;
98          }
99          assertTrue("Highest acceleration should be less than max", referenceAcceleration <= 1.25);
100         // Check that the returned acceleration increases with the speed of the leader
101         // System.out.println("");
102         referenceAcceleration = Double.NEGATIVE_INFINITY;
103         leaderPosition =
104                 new DoubleScalar.Abs<LengthUnit>(2 + 3 + referenceCar.length().getValueSI()
105                         + referenceCar.getPosition(initialTime).getValueSI(), LengthUnit.METER);
106         // In IDM+ the reference car must have non-zero speed for the leader speed to have any effect
107         initialSpeed = new DoubleScalar.Rel<SpeedUnit>(2, SpeedUnit.METER_PER_SECOND);
108         for (int integerLeaderSpeed = 0; integerLeaderSpeed <= 40; integerLeaderSpeed++)
109         {
110             referenceCar = new Car(12345, simulator, carFollowingModel, initialTime, initialPosition, initialSpeed);
111             leaders.clear();
112             DoubleScalar.Rel<SpeedUnit> leaderSpeed =
113                     new DoubleScalar.Rel<SpeedUnit>(integerLeaderSpeed, SpeedUnit.METER_PER_SECOND);
114             leaderCar = new Car(0, simulator, null, initialTime, leaderPosition, leaderSpeed);
115             leaders.add(leaderCar);
116             // System.out.println("referenceCar: " + referenceCar);
117             // System.out.println("leaderCar   : " + leaderCar);
118             cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
119             double acceleration = cfmr.acceleration.getValueSI();
120             // System.out.println("Acceleration with leader driving " + integerLeaderSpeed + " m/s is " + acceleration);
121             assertTrue("acceleration should not decrease when leader speed is increased", acceleration >= referenceAcceleration);
122             referenceAcceleration = acceleration;
123         }
124         assertTrue("Highest acceleration should be less than max", referenceAcceleration <= 1.25);
125         // Check that a car that is 100m behind a stationary car accelerates, then decelerates and stops at the right
126         // point. (In IDM+ the car oscillates a while around the final position with pretty good damping.)
127         initialPosition = new DoubleScalar.Abs<LengthUnit>(100, LengthUnit.METER);
128         initialSpeed = new DoubleScalar.Rel<SpeedUnit>(0, SpeedUnit.METER_PER_SECOND);
129         referenceCar = new Car(12345, simulator, carFollowingModel, initialTime, initialPosition, initialSpeed);
130         leaders.clear();
131         leaderPosition =
132                 new DoubleScalar.Abs<LengthUnit>(100 + 3 + referenceCar.length().getValueSI()
133                         + referenceCar.getPosition(initialTime).getValueSI(), LengthUnit.METER);
134         leaderCar = new Car(0, simulator, null, initialTime, leaderPosition, initialSpeed);
135         leaders.add(leaderCar);
136         // System.out.println("Setup    referenceCar: " + referenceCar);
137         for (int timeStep = 0; timeStep < 200; timeStep++)
138         {
139             cfmr = carFollowingModel.computeAcceleration(referenceCar, leaders, speedLimit);
140             referenceCar.setState(cfmr);
141             // System.out.println(String.format("step %3d referenceCar: %s", timeStep, referenceCar));
142             if (timeStep > 100)
143             {
144                 double position = referenceCar.getPosition(cfmr.validUntil).getValueSI();
145                 assertEquals("After 20 seconds the referenceCar should now be very close to 3m before the rear of the leader",
146                         200, position, 0.1);
147                 assertEquals("After 20 seconds the speed of the referenceCar should be almost 0", 0,
148                         referenceCar.getVelocity(cfmr.validUntil).getValueSI(), 0.2);
149             }
150         }
151     }
