19  public class CarTest
20  {
21      /**
22       * Test some basics of the Car class.
23       */
24      @SuppressWarnings("static-method")
25      @Test
26      public final void carTest()
27      {
28          DoubleScalar.Abs<TimeUnit> initialTime = new DoubleScalar.Abs<TimeUnit>(0, TimeUnit.SECOND);
29          DoubleScalar.Abs<LengthUnit> initialPosition = new DoubleScalar.Abs<LengthUnit>(12, LengthUnit.METER);
30          DoubleScalar.Rel<SpeedUnit> initialSpeed = new DoubleScalar.Rel<SpeedUnit>(34, SpeedUnit.KM_PER_HOUR);
31          Car referenceCar = new Car(12345, null, null, initialTime, initialPosition, initialSpeed);
32          assertEquals("The car should store it's ID", 12345, (int) referenceCar.getID());
33          assertEquals("At t=initialTime the car should be at it's initial position", initialPosition.getValueSI(), referenceCar
34                  .getPosition(initialTime).getValueSI(), 0.0001);
35          assertEquals("The car should store it's initial speed", initialSpeed.getValueSI(), referenceCar
36                  .getVelocity(initialTime).getValueSI(), 0.00001);
37          assertEquals("The car should have an initial acceleration equal to 0", 0, referenceCar.acceleration.getValueSI(),
38                  0.0001);
39      }
