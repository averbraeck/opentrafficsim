34       * Test the FundamentalDiagram.
35       */
36      @SuppressWarnings("static-method")
37      @Test
38      public final void fundamentalDiagramTest()
39      {
40          DoubleScalar.Rel<TimeUnit> aggregationTime = new DoubleScalar.Rel<TimeUnit>(30, TimeUnit.SECOND);
41          DoubleScalar.Abs<LengthUnit> position = new DoubleScalar.Abs<LengthUnit>(123, LengthUnit.METER);
42          FundamentalDiagram fd;
43          for (int numberOfLanes = -2; numberOfLanes <= 0; numberOfLanes++)
44          {
45              try
46              {
47                  fd = new FundamentalDiagram("Fundamental Diagram", -1, aggregationTime, position);
48                  fail("Bad number of lanes should have thrown an Error");
49              }
50              catch (Error e)
51              {
52                  // Ignore
53              }
54          }
55          try
56          {
57              fd = new FundamentalDiagram("Fundamental Diagram", 1, new DoubleScalar.Rel<TimeUnit>(0, TimeUnit.SECOND), position);
58              fail("Bad number of lanes should have thrown an Error");
59          }
60          catch (Error e)
61          {
62              // Ignore
63          }
64          for (int numberOfLanes = 1; numberOfLanes <= 3; numberOfLanes++)
65          {
66              fd = new FundamentalDiagram("Fundamental Diagram", numberOfLanes, aggregationTime, position);
67              assertEquals("SeriesCount should match numberOfLanes", numberOfLanes, fd.getSeriesCount());
68              assertEquals("Position should match the supplied position", position.getValueSI(), fd.getPosition().getValueSI(),
69                      0.0001);
70              try
71              {
72                  fd.getXValue(-1, 0);
73                  fail("Bad series should have thrown an Error");
74              }
75              catch (Error e)
76              {
77                  // Ignore
78              }
79              try
80              {
81                  fd.getXValue(numberOfLanes, 0);
82                  fail("Bad series should have thrown an Error");
83              }
84              catch (Error e)
85              {
86                  // Ignore
87              }
88              double value = fd.getXValue(0, 0);
89              assertTrue("No data should result in NaN", Double.isNaN(value));
90              value = fd.getX(0, 0).doubleValue();
91              assertTrue("No data should result in NaN", Double.isNaN(value));
92              value = fd.getYValue(0, 0);
93              assertTrue("No data should result in NaN", Double.isNaN(value));
94              value = fd.getY(0, 0).doubleValue();
95              assertTrue("No data should result in NaN", Double.isNaN(value));
96              ActionEvent setXToSpeed = new ActionEvent(fd, 0, "Speed/Speed");
97              ActionEvent resetAxis = new ActionEvent(fd, 0, "Flow/Density");
98              DoubleScalar.Rel<SpeedUnit> speed = new DoubleScalar.Rel<SpeedUnit>(100, SpeedUnit.KM_PER_HOUR);
99              DoubleScalar.Abs<TimeUnit> time = new DoubleScalar.Abs<TimeUnit>(123, TimeUnit.SECOND);
100             int bucket = (int) Math.floor(time.getValueSI() / aggregationTime.getValueSI());
101             for (int lane = 0; lane < numberOfLanes; lane++)
102             {
103                 Car car = new Car(1 + lane, null, null, time, new DoubleScalar.Abs<LengthUnit>(23, LengthUnit.METER), speed);
104                 fd.addData(lane, car, time);
105                 for (int readBackLane = 0; readBackLane < numberOfLanes; readBackLane++)
106                 {
107                     for (int sample = 0; sample < 10; sample++)
108                     {
109                         boolean shouldHaveData = readBackLane <= lane && sample == bucket;
110                         value = fd.getXValue(readBackLane, sample);
111                         if (shouldHaveData)
112                         {
113                             double expectedDensity = 3600 / aggregationTime.getValueSI() / speed.getValueSI();
114                             assertEquals("Density should be " + expectedDensity, expectedDensity, value, 0.00001);
115                         }
116                         else
117                         {
118                             assertTrue("Data should be NaN", Double.isNaN(value));
119                         }
120                         value = fd.getX(readBackLane, sample).doubleValue();
121                         if (shouldHaveData)
122                         {
123                             double expectedDensity = 3600 / aggregationTime.getValueSI() / speed.getValueSI();
124                             assertEquals("Density should be " + expectedDensity, expectedDensity, value, 0.00001);
125                         }
126                         else
127                         {
128                             assertTrue("Data should be NaN", Double.isNaN(value));
129                         }
130                         shouldHaveData = readBackLane <= lane && sample <= bucket;
131                         value = fd.getYValue(readBackLane, sample);
132                         if (shouldHaveData)
133                         {
134                             double expectedFlow =
135                                     readBackLane <= lane && sample == bucket ? 3600 / aggregationTime.getValueSI() : 0;
136                             assertEquals("Flow should be " + expectedFlow, expectedFlow, value, 0.00001);
137                         }
138                         else
139                         {
140                             assertTrue("Data should be NaN", Double.isNaN(value));
141                         }
142                         value = fd.getY(readBackLane, sample).doubleValue();
143                         if (shouldHaveData)
144                         {
145                             double expectedFlow =
146                                     readBackLane <= lane && sample == bucket ? 3600 / aggregationTime.getValueSI() : 0;
147                             assertEquals("Flow should be " + expectedFlow, expectedFlow, value, 0.00001);
148                         }
149                         else
150                         {
151                             assertTrue("Data should be NaN", Double.isNaN(value));
152                         }
153                         fd.actionPerformed(setXToSpeed);
154                         value = fd.getYValue(readBackLane, sample);
155                         if (shouldHaveData)
156                         {
157                             double expectedSpeed = readBackLane <= lane && sample == bucket ? speed.getValueInUnit() : 0;
158                             assertEquals("Speed should be " + expectedSpeed, expectedSpeed, value, 0.00001);
159                         }
160                         else
161                         {
162                             assertTrue("Data should be NaN", Double.isNaN(value));
163                         }
164                         value = fd.getY(readBackLane, sample).doubleValue();
165                         if (shouldHaveData)
166                         {
167                             double expectedSpeed = readBackLane <= lane && sample == bucket ? speed.getValueInUnit() : 0;
168                             assertEquals("Speed should be " + expectedSpeed, expectedSpeed, value, 0.00001);
169                         }
170                         else
171                         {
172                             assertTrue("Data should be NaN", Double.isNaN(value));
173                         }
174                         fd.actionPerformed(resetAxis);
175                     }
176                 }
177             }
178             // Check that harmonic mean speed is computed
179             speed = new DoubleScalar.Rel<SpeedUnit>(10, SpeedUnit.KM_PER_HOUR);
180             Car car = new Car(1234, null, null, time, new DoubleScalar.Abs<LengthUnit>(23, LengthUnit.METER), speed);
181             fd.addData(0, car, time);
182             fd.actionPerformed(setXToSpeed);
183             value = fd.getYValue(0, bucket);
184             double expected = 2d / (1d / 100 + 1d / 10);
185             // System.out.println("harmonic speed is " + value + ", expected is " + expected);
186             assertEquals("Harmonic mean of 10 and 100 is " + expected, expected, value, 0.0001);
187             // Test the actionPerformed method with various malformed ActionEvents.
188             try
189             {
190                 fd.actionPerformed(new ActionEvent(fd, 0, "bla"));
191                 fail("Bad ActionEvent should have thrown an Error");
192             }
193             catch (Error e)
194             {
195                 // Ignore
196             }
197             try
198             {
199                 fd.actionPerformed(new ActionEvent(fd, 0, "Speed/bla"));
200                 fail("Bad ActionEvent should have thrown an Error");
201             }
202             catch (Error e)
203             {
204                 // Ignore
205             }
206             try
207             {
208                 fd.actionPerformed(new ActionEvent(fd, 0, "Flow/bla"));
209                 fail("Bad ActionEvent should have thrown an Error");
210             }
211             catch (Error e)
212             {
213                 // Ignore
214             }
215             try
216             {
217                 fd.actionPerformed(new ActionEvent(fd, 0, "Density/bla"));
218                 fail("Bad ActionEvent should have thrown an Error");
219             }
220             catch (Error e)
221             {
222                 // Ignore
223             }

226                 fd.actionPerformed(new ActionEvent(fd, 0, "bla/Speed"));
227                 fail("Bad ActionEvent should have thrown an Error");
228             }

254     /**
255      * Test the updateHint method in the PointerHandler.
256      */
257     @SuppressWarnings("static-method")
258     @Test
259     public final void testHints()
260     {
261         DoubleScalar.Rel<TimeUnit> aggregationTime = new DoubleScalar.Rel<TimeUnit>(30, TimeUnit.SECOND);
262         DoubleScalar.Abs<LengthUnit> position = new DoubleScalar.Abs<LengthUnit>(123, LengthUnit.METER);
263         FundamentalDiagram fd = new FundamentalDiagram("Fundamental Diagram", 1, aggregationTime, position);
264         // First get the panel that stores the result of updateHint (this is ugly)
265         JLabel hintPanel = null;
266         ChartPanel chartPanel = null;
267         for (Component c0 : fd.getComponents())
268         {
269             for (Component c1 : ((Container) c0).getComponents())
270             {
271                 if (c1 instanceof Container)
272                 {
273                     for (Component c2 : ((Container) c1).getComponents())
274                     {
275                         // System.out.println("c2 is " + c2);
276                         if (c2 instanceof Container)
277                         {
278                             for (Component c3 : ((Container) c2).getComponents())
279                             {
280                                 // System.out.println("c3 is " + c3);
281                                 if (c3 instanceof JLabel)
282                                 {
283                                     if (null == hintPanel)
284                                     {
285                                         hintPanel = (JLabel) c3;
286                                     }
287                                     else
288                                     {
289                                         fail("There should be only one JPanel in a FundamentalDiagram");
290                                     }
291                                 }
292                                 if (c3 instanceof ChartPanel)
293                                 {
294                                     if (null == chartPanel)
295                                     {
296                                         chartPanel = (ChartPanel) c3;
297                                     }
298                                     else
299                                     {
300                                         fail("There should be only one ChartPanel in a FundamentalDiagram");
301                                     }
302                                 }
303                             }
304                         }
305                     }
306                 }
307             }
308         }
309         if (null == hintPanel)
310         {
311             fail("Could not find a JLabel in FundamentalDiagram");
312         }
313         if (null == chartPanel)
314         {
315             fail("Could not find a ChartPanel in FundamentalDiagram");
316         }
317         assertEquals("Initially the text should be a single space", " ", hintPanel.getText());
318         PointerHandler ph = null;
319         for (MouseListener ml : chartPanel.getMouseListeners())
320         {
321             if (ml instanceof PointerHandler)
322             {
323                 if (null == ph)
324                 {
325                     ph = (PointerHandler) ml;
326                 }
327                 else
328                 {
329                     fail("There should be only one PointerHandler on the chartPanel");
330                 }
331             }
332         }
333         if (null == ph)
334         {
335             fail("Could not find the PointerHandler for the chartPanel");
336         }
337         ph.updateHint(1, 2);
338         // System.out.println("Hint text is now " + hintPanel.getText());
339         assertFalse("Hint should not be a single space", " ".equals(hintPanel.getText()));
340         ph.updateHint(Double.NaN, Double.NaN);
341         assertEquals("The text should again be a single space", " ", hintPanel.getText());
342     }
