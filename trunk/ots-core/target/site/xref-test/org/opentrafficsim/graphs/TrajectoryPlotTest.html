36      /** Sample interval for the TrajectoryPlot. */
37      DoubleScalar.Rel<TimeUnit> sampleInterval = new DoubleScalar.Rel<TimeUnit>(0.25, TimeUnit.SECOND);
38  
39      /**
40       * Test the TrajectoryPlot.
41       */
42      @Test
43      public final void trajectoryTest()
44      {
45          DoubleScalar.Abs<LengthUnit> minimumDistance = new DoubleScalar.Abs<LengthUnit>(1234, LengthUnit.METER);
46          DoubleScalar.Abs<LengthUnit> maximumDistance = new DoubleScalar.Abs<LengthUnit>(12345, LengthUnit.METER);
47  
48          TrajectoryPlot tp = new TrajectoryPlot("Trajectory", this.sampleInterval, minimumDistance, maximumDistance);
49          assertTrue("newly created DensityContourPlot should not be null", null != tp);
50          assertEquals("Number of trajectories should initially be 0", 0, tp.getSeriesCount());
51          for (int i = -10; i <= 10; i++)
52          {
53              assertEquals("SeriesKey(" + i + ") should return " + i, i, tp.getSeriesKey(i));
54          }
55          assertEquals("Domain order should be ASCENDING", DomainOrder.ASCENDING, tp.getDomainOrder());
56          DoubleScalar.Abs<TimeUnit> initialTime = new DoubleScalar.Abs<TimeUnit>(100, TimeUnit.SECOND);
57          DoubleScalar.Abs<LengthUnit> initialPosition = new DoubleScalar.Abs<LengthUnit>(2000, LengthUnit.METER);
58          DoubleScalar.Rel<SpeedUnit> initialSpeed = new DoubleScalar.Rel<SpeedUnit>(50, SpeedUnit.KM_PER_HOUR);
59          // Create a car running 50 km.h
60          Car car = new Car(1, null, null, initialTime, initialPosition, initialSpeed);
61          // Make the car accelerate with constant acceleration of 0.05 m/s/s for 500 seconds
62          DoubleScalar.Abs<TimeUnit> endTime = new DoubleScalar.Abs<TimeUnit>(initialTime.getValueSI() + 400, TimeUnit.SECOND);
63          car.setState(new CarFollowingModelResult(new DoubleScalar.Abs<AccelerationUnit>(0.05,
64                  AccelerationUnit.METER_PER_SECOND_2), endTime, 0));
65          // System.out.println("Car end position " + car.getPosition(car.getNextEvaluationTime()));
66          tp.addData(car);
67          assertEquals("Number of trajectories should now be 1", 1, tp.getSeriesCount());
68          verifyTrajectory(car, 0, tp);
69          initialTime = new DoubleScalar.Abs<TimeUnit>(150, TimeUnit.SECOND);
70          Car secondCar = new Car(2, null, null, initialTime, initialPosition, initialSpeed);
71          // Make the second car accelerate with constant acceleration of 0.03 m/s/s for 500 seconds
72          secondCar.setState(new CarFollowingModelResult(new DoubleScalar.Abs<AccelerationUnit>(0.03,
73                  AccelerationUnit.METER_PER_SECOND_2), endTime, 0));
74          // System.out.println("Second car end position " + car.getPosition(secondCar.getNextEvaluationTime()));
75          tp.addData(secondCar);
76          assertEquals("Number of trajectories should now be 2", 2, tp.getSeriesCount());
77          verifyTrajectory(car, 0, tp); // first car trajectory should not change by adding the second
78          verifyTrajectory(secondCar, 1, tp);
79          // Check the updateHint method in the PointerHandler
80          // First get the panel that stores the result of updateHint (this is ugly)
81          JLabel hintPanel = null;
82          ChartPanel chartPanel = null;
83          for (Component c0 : tp.getComponents())
84          {
85              for (Component c1 : ((Container) c0).getComponents())
86              {
87                  if (c1 instanceof Container)
88                  {
89                      for (Component c2 : ((Container) c1).getComponents())
90                      {
91                          // System.out.println("c2 is " + c2);
92                          if (c2 instanceof Container)
93                          {
94                              for (Component c3 : ((Container) c2).getComponents())
95                              {
96                                  // System.out.println("c3 is " + c3);
97                                  if (c3 instanceof JLabel)
98                                  {
99                                      if (null == hintPanel)
100                                     {
101                                         hintPanel = (JLabel) c3;
102                                     }
103                                     else
104                                     {
105                                         fail("There should be only one JPanel in a ContourPlot");
106                                     }
107                                 }
108                                 if (c3 instanceof ChartPanel)
109                                 {
110                                     if (null == chartPanel)
111                                     {
112                                         chartPanel = (ChartPanel) c3;
113                                     }
114                                     else
115                                     {
116                                         fail("There should be only one ChartPanel in a ContourPlot");
117                                     }
118                                 }
119                             }
120                         }
121                     }
122                 }
123             }
124         }
125         if (null == hintPanel)
126         {
127             fail("Could not find a JLabel in ContourPlot");
128         }
129         if (null == chartPanel)
130         {
131             fail("Could not find a ChartPanel in ContourPlot");
132         }
133         assertEquals("Initially the text should be a single space", " ", hintPanel.getText());
134         PointerHandler ph = null;
135         for (MouseListener ml : chartPanel.getMouseListeners())
136         {
137             if (ml instanceof PointerHandler)
138             {
139                 if (null == ph)
140                 {
141                     ph = (PointerHandler) ml;
142                 }
143                 else
144                 {
145                     fail("There should be only one PointerHandler on the chartPanel");
146                 }
147             }
148         }
149         if (null == ph)
150         {
151             fail("Could not find the PointerHandler for the chartPanel");
152         }
153         ph.updateHint(1, 2);
154         // System.out.println("Hint text is now " + hintPanel.getText());
155         assertFalse("Hint should not be a single space", " ".equals(hintPanel.getText()));
156         ph.updateHint(Double.NaN, Double.NaN);
157         assertEquals("The text should again be a single space", " ", hintPanel.getText());
158     }

160     /**
161      * Verify that a sampled trajectory matches the actual trajectory.
162      * @param car Car; the car whose trajectory was sampled
163      * @param series Integer; the series in the TrajectoryPlot that should correspond to the car
164      * @param tp TrajectoryPlot; the TrajectoryPlot that contains the samples
165      */
166     private void verifyTrajectory(final Car car, final int series, final TrajectoryPlot tp)
167     {
168         DoubleScalar.Abs<TimeUnit> initialTime = car.getLastEvaluationTime();
169         DoubleScalar.Rel<TimeUnit> duration =
170                 MutableDoubleScalar.minus(car.getNextEvaluationTime(), car.getLastEvaluationTime()).immutable();
171         int expectedNumberOfSamples = (int) (duration.getValueSI() / this.sampleInterval.getValueSI());
172         assertEquals("Number of samples in trajectory should be ", expectedNumberOfSamples, tp.getItemCount(series));
173         // Check that the stored trajectory accurately matches the trajectory of the car at all sampling times
174         for (int sample = 0; sample < expectedNumberOfSamples; sample++)
175         {
176             DoubleScalar.Rel<TimeUnit> deltaTime =
177                     new DoubleScalar.Rel<TimeUnit>(this.sampleInterval.getValueSI() * sample, TimeUnit.SECOND);
178             DoubleScalar.Abs<TimeUnit> sampleTime = MutableDoubleScalar.plus(initialTime, deltaTime).immutable();
179             double sampledTime = tp.getXValue(series, sample);
180             assertEquals("Sample should have been taken at " + sampleTime, sampleTime.getValueSI(), sampledTime, 0.0001);
181             sampledTime = tp.getX(series, sample).doubleValue();
182             assertEquals("Sample should have been taken at " + sampleTime, sampleTime.getValueSI(), sampledTime, 0.0001);
183             DoubleScalar.Abs<LengthUnit> actualPosition = car.getPosition(sampleTime);
184             double sampledPosition = tp.getYValue(series, sample);
185             assertEquals("Sample position should have been " + actualPosition, actualPosition.getValueSI(), sampledPosition,
186                     0.0001);
187             sampledPosition = tp.getY(series, sample).doubleValue();
188             assertEquals("Sample position should have been " + actualPosition, actualPosition.getValueSI(), sampledPosition,
189                     0.0001);
190         }
191     }
