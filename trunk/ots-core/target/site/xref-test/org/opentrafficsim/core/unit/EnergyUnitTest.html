19  public class EnergyUnitTest extends AbstractUnitTest<EnergyUnit>
20  {
21      /**
22       * Set the locale to "en" so we know what texts should be retrieved from the resources.
23       */
24      @SuppressWarnings("static-method")
25      @Before
26      public final void setup()
27      {
28          DefaultLocale.setLocale(new Locale("en"));
29      }
30  
31      /**
32       * Verify the result of some get*Key methods.
33       */
34      @Test
35      public final void keys()
36      {
37          checkKeys(EnergyUnit.JOULE, "EnergyUnit.Joule", "EnergyUnit.J");
38      }
39  
40      /**
41       * Verify conversion factors, English names and abbreviations.
42       */
43      @Test
44      public final void conversions()
45      {
46          checkUnitRatioNameAndAbbreviation(EnergyUnit.JOULE, 1, 0.00000001, "Joule", "J");
47          checkUnitRatioNameAndAbbreviation(EnergyUnit.FOOT_POUND_FORCE, 1.35581794833, 0.0000005, "foot pound-force", "ft.lbf");
48          checkUnitRatioNameAndAbbreviation(EnergyUnit.BTU_ISO, 1054.5, 0.001, "British thermal unit (ISO)", "BTU(ISO)");
49          // Check two conversions between non-standard units
50          assertEquals("one FOOT POUND FORCE is about 0.0013 BTU ISO", 0.0013,
51                  getMultiplicationFactorTo(EnergyUnit.FOOT_POUND_FORCE, EnergyUnit.BTU_ISO), 0.0001);
52          // Check conversion factor to standard unit for all remaining acceleration units
53          checkUnitRatioNameAndAbbreviation(EnergyUnit.INCH_POUND_FORCE, 0.112984829, 0.000000001, "inch pound-force", "in.lbf");
54          checkUnitRatioNameAndAbbreviation(EnergyUnit.BTU_IT, 1055.05585262, 0.000001,
55                  "British thermal unit (International Table)", "BTU(IT)");
56          checkUnitRatioNameAndAbbreviation(EnergyUnit.CALORIE_IT, 4.1868, 0.00005, "calorie (International Table)", "cal(IT)");
57          checkUnitRatioNameAndAbbreviation(EnergyUnit.KILOCALORIE, 4186.8, 0.05, "kilocalorie", "kcal");
58          checkUnitRatioNameAndAbbreviation(EnergyUnit.KILOWATT_HOUR, 3600000, 0.1, "kilowatt-hour", "kWh");
59      }
