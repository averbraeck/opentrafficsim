22  {
23      /**
24       * Set the locale to "en" so we know what texts should be retrieved from the resources.
25       */
26      @SuppressWarnings("static-method")
27      @Before
28      public final void setup()
29      {
30          DefaultLocale.setLocale(new Locale("en"));
31      }

33      /**
34       * Verify the result of some get*Key methods.
35       */
36      @Test
37      public final void keys()
38      {
39          checkKeys(AngleSolidUnit.STERADIAN, "AngleSolidUnit.steradian", "AngleSolidUnit.sr");
40      }
41  
42      /**
43       * Verify conversion factors, English names and abbreviations.
44       */
45      @Test
46      public final void conversions()
47      {
48          checkUnitRatioNameAndAbbreviation(AngleSolidUnit.STERADIAN, 1, 0.0000001, "steradian", "sr");
49          checkUnitRatioNameAndAbbreviation(AngleSolidUnit.SQUARE_DEGREE, 1.0 / 3283, 0.0005, "square degree", "sq.deg");
50          // Check two conversions between units
51          assertEquals("one STERADIAN is about 3283 SQUARE_DEGREE", 3283,
52                  getMultiplicationFactorTo(AngleSolidUnit.STERADIAN, AngleSolidUnit.SQUARE_DEGREE), 0.5);
53          assertEquals("one SQUARE_DEGREE is about 0.0003045 STERADIAN", 0.0003045,
54                  getMultiplicationFactorTo(AngleSolidUnit.SQUARE_DEGREE, AngleSolidUnit.STERADIAN), 0.0000005);
55      }
56  
57      /**
58       * Verify that we can create our own angle unit.
59       */
60      @Test
61      public final void createAngleUnit()
62      {
63          AngleSolidUnit myAPU =
64                  new AngleSolidUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "AngleSolidUnit.point",
65                          UnitLocalizationsTest.DONOTCHECKPREFIX + "AngleSolidUnit.pt", OTHER, AngleSolidUnit.STERADIAN,
66                          0.19634954085);
67          assertTrue("Can create a new TimeUnit", null != myAPU);
68          checkUnitRatioNameAndAbbreviation(myAPU, 0.19634954085, 0.0000001, "!point!", "!pt!");
69      }
