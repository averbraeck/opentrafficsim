49      /**
50       * Check that the values in a DoubleMatrix match with the values in a 2D double array.
51       * @param fm DoubleMatrix; the DoubleMatrix to match
52       * @param reference double[][]; the 2D double array
53       * @param precision double; the maximum allowed error
54       * @param expectAbsolute boolean; if true; the DoubleMatrix should be Absolute; otherwise it should be Relative
55       * @param expectDense boolean; if true; the DoubleMatrix should be Dense; otherwise it should be Sparse
56       */
57      private static void checkContentsAndType(final DoubleMatrix<?> fm, final double[][] reference, final double precision,
58              final boolean expectAbsolute, final boolean expectDense)
59      {
60          assertTrue("DoubleMatrix argument should not be null", null != fm);
61          assertEquals("Number of rows should match", fm.rows(), reference.length);
62          assertEquals("Number of columns should match", fm.columns(), reference[0].length);
63          for (int row = 0; row < fm.rows(); row++)
64          {
65              for (int column = 0; column < fm.columns(); column++)
66              {
67                  try
68                  {
69                      assertEquals("Value should match", reference[row][column], fm.getInUnit(row, column), precision);
70                  }
71                  catch (ValueException exception)
72                  {
73                      fail("Unexpected exception");
74                  }
75              }
76          }
77          if (expectAbsolute)
78          {
79              assertTrue("DoubleMatrix argument should be Absolute", fm instanceof Absolute);
80          }
81          else
82          {
83              assertTrue("DoubleMatrix argument should be Relative", fm instanceof Relative);
84          }
85          if (expectDense)
86          {
87              assertTrue("DoubleMatrix argument should be Dense", fm instanceof DenseData);
88          }
89          else
90          {
91              assertTrue("DoubleMatrix argument should be Sparse", fm instanceof SparseData);
92          }
93          MutableDoubleMatrix<?> mutableVersion = fm.mutable();
94          try
95          {
96              mutableVersion.setSI(0, 0, mutableVersion.getSI(0, 0) + 10);
97              assertTrue("Value in mutable version is changed", mutableVersion.getSI(0, 0) != fm.getSI(0, 0));
98          }
99          catch (ValueException exception)
100         {
101             fail("Unexpected exception");
102         }
103         for (int row = 0; row < fm.rows(); row++)
104         {
105             for (int column = 0; column < fm.columns(); column++)
106             {
107                 try
108                 {
109                     assertEquals("Value should still match", reference[row][column], fm.getInUnit(row, column), precision);
110                 }
111                 catch (ValueException exception)
112                 {
113                     fail("Unexpected exception");
114                 }
115             }
116         }
117         if (fm instanceof MutableDoubleMatrix)
118         {
119             MutableDoubleMatrix<?> mfm = (MutableDoubleMatrix<?>) fm;
120             DoubleMatrix<?> immutableVersion = mfm.immutable();
121             try
122             {
123                 mfm.setSI(0, 0, mfm.getSI(0, 0) + 10);
124                 assertTrue("Only the value in the mutable version is changed", mfm.getSI(0, 0) != reference[0][0]);
125                 assertEquals("Value in the immutable version still matches the reference", reference[0][0],
126                         immutableVersion.getSI(0, 0), precision);
127             }
128             catch (ValueException exception)
129             {
130                 fail("Unexpected exception");
131             }
132         }
133     }

312     /**
313      * Create a 2D array of double for testing. Entry 0,0 is zero, all others will be non-zero unless a non-zero offset is
314      * specified.
315      * @param rows Integer; number of rows in the 2D array
316      * @param cols Integer; number of columns in the 2D array
317      * @param badRow Boolean; if true; make the last row one entry longer than the rest
318      * @param offset Double; number to add to each entry
319      * @return double[][]; 2D array of double
320      */
321     private static double[][] buildArray(final int rows, final int cols, final boolean badRow, final double offset)
322     {
323         double[][] result = new double[rows][];
324         int badRowIndex = badRow ? rows - 1 : -1;
325         for (int row = 0; row < rows; row++)
326         {
327             result[row] = new double[row == badRowIndex ? cols + 1 : cols];
328             for (int col = 0; col < result[row].length; col++)
329             {
330                 result[row][col] = row * 1000 + col + offset;
331             }
332         }
333         return result;
334     }
