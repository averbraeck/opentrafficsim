40      }
41  
42      /**
43       * Verify conversion factors, English names and abbreviations.
44       */
45      @Test
46      public final void conversions()
47      {
48          checkUnitRatioNameAndAbbreviation(MassUnit.KILOGRAM, 1, 0.00000001, "kilogram", "kg");
49          checkUnitRatioNameAndAbbreviation(MassUnit.GRAM, 0.001, 0.000000001, "gram", "g");
50          checkUnitRatioNameAndAbbreviation(MassUnit.POUND, 0.453592, 0.000001, "pound", "lb");
51          // Check two conversions between non-standard units
52          assertEquals("one POUND is 16 OUNCE", 16, getMultiplicationFactorTo(MassUnit.POUND, MassUnit.OUNCE), 0.00001);
53          assertEquals("one OUNCE is 0.0625 POUND", 0.0625, getMultiplicationFactorTo(MassUnit.OUNCE, MassUnit.POUND), 0.000001);
54          // Check conversion factor to standard unit for all remaining time units
55          checkUnitRatioNameAndAbbreviation(MassUnit.OUNCE, 0.0283495, 0.0000001, "ounce", "oz");
56          checkUnitRatioNameAndAbbreviation(MassUnit.TON_LONG, 1016.046906, 0.00001, "long ton", "long tn");
57          checkUnitRatioNameAndAbbreviation(MassUnit.TON_SHORT, 907.18474, 0.00001, "short ton", "sh tn");
58          checkUnitRatioNameAndAbbreviation(MassUnit.TON_METRIC, 1000, 0.001, "metric tonne", "t");
59      }
60  
61      /**
62       * Verify that we can create our own mass unit.
63       */
64      @Test
65      public final void createMassUnit()
66      {
67          MassUnit myMU =
68                  new MassUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "MassUnit.Person", UnitLocalizationsTest.DONOTCHECKPREFIX
69                          + "MassUnit.pn", OTHER, MassUnit.KILOGRAM, 80);
70          assertTrue("Can create a new MassUnit", null != myMU);
71          checkUnitRatioNameAndAbbreviation(myMU, 80, 1, "!Person!", "!pn!");
72      }
