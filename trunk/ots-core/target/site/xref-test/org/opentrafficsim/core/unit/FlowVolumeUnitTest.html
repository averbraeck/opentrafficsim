22  public class FlowVolumeUnitTest extends AbstractUnitTest<FlowVolumeUnit>
23  {
24      /**
25       * Set the locale to "en" so we know what texts should be retrieved from the resources.
26       */
27      @SuppressWarnings("static-method")
28      @Before
29      public final void setup()
30      {
31          DefaultLocale.setLocale(new Locale("en"));
32      }
33  
34      /**
35       * Verify the result of some get*Key methods.
36       */
37      @Test
38      public final void keys()
39      {
40          checkKeys(FlowVolumeUnit.CUBIC_METER_PER_SECOND, "FlowVolumeUnit.cubic_meter_per_second",
41                  "FlowVolumeUnit.m^3/s");
42      }
43  
44      /**
45       * Verify conversion factors, English names and abbreviations.
46       */
47      @Test
48      public final void conversions()
49      {
50          checkUnitRatioNameAndAbbreviation(FlowVolumeUnit.CUBIC_METER_PER_SECOND, 1, 0.000001, "cubic meter per second",
51                  "m^3/s");
52          checkUnitRatioNameAndAbbreviation(FlowVolumeUnit.CUBIC_METER_PER_MINUTE, 0.0166667, 0.000001,
53                  "cubic meter per minute", "m^3/min");
54          // Check two conversions between non-standard units
55          assertEquals("one CUBIC METER PER HOUR is about 2.205 CUBIC_METER_PER_MINUTED", 0.01666667,
56                  getMultiplicationFactorTo(FlowVolumeUnit.CUBIC_METER_PER_HOUR, FlowVolumeUnit.CUBIC_METER_PER_MINUTE),
57                  0.00001);
58          assertEquals("one CUBIC METER PER MINUTE is 60 CUBIC_METER_PER_HOUR", 60,
59                  getMultiplicationFactorTo(FlowVolumeUnit.CUBIC_METER_PER_MINUTE, FlowVolumeUnit.CUBIC_METER_PER_HOUR),
60                  0.0001);
61      }
62  
63      /**
64       * Verify that we can create our own FlowVolume unit.
65       */
66      @Test
67      public final void createFLowVolumeUnit()
68      {
69          FlowVolumeUnit myFVU =
70                  new FlowVolumeUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "FlowVolumeUnit.TrucksPerHour",
71                          UnitLocalizationsTest.DONOTCHECKPREFIX + "FlowVolumeUnit.tph", UnitSystem.OTHER,
72                          FlowVolumeUnit.CUBIC_METER_PER_HOUR, 100);
73          assertTrue("Can create a new FlowMassUnit", null != myFVU);
74          checkUnitRatioNameAndAbbreviation(myFVU, 100. / 3600, 0.0001, "!TrucksPerHour!", "!tph!");
75      }
