22  public class ElectricalPotentialUnitTest extends AbstractUnitTest<ElectricalPotentialUnit>
23  {
24      /**
25       * Set the locale to "en" so we know what texts should be retrieved from the resources.
26       */
27      @SuppressWarnings("static-method")
28      @Before
29      public final void setup()
30      {
31          DefaultLocale.setLocale(new Locale("en"));
32      }
33  
34      /**
35       * Verify the result of some get*Key methods.
36       */
37      @Test
38      public final void keys()
39      {
40          checkKeys(ElectricalPotentialUnit.VOLT, "ElectricalPotentialUnit.volt", "ElectricalPotentialUnit.V");
41      }
42  
43      /**
44       * Verify conversion factors, English names and abbreviations.
45       */
46      @Test
47      public final void conversions()
48      {
49          checkUnitRatioNameAndAbbreviation(ElectricalPotentialUnit.VOLT, 1, 0.00000001, "volt", "V");
50          checkUnitRatioNameAndAbbreviation(ElectricalPotentialUnit.MILLIVOLT, 0.001, 0.00000000001, "millivolt", "mV");
51          checkUnitRatioNameAndAbbreviation(ElectricalPotentialUnit.KILOVOLT, 1000, 0.005, "kilovolt", "kV");
52          // Check two conversions between non-standard units
53          assertEquals("one KILOVOLT is 1000000 MILLIVOLT", 1000000,
54                  getMultiplicationFactorTo(ElectricalPotentialUnit.KILOVOLT, ElectricalPotentialUnit.MILLIVOLT), 0.0001);
55      }
56  
57      /**
58       * Verify that we can create our own electrical potential unit.
59       */
60      @Test
61      public final void createElectricalPotentialUnit()
62      {
63          ElectricalPotentialUnit myEPU =
64                  new ElectricalPotentialUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "ElectricalPotentialUnit.NanoVolt",
65                          UnitLocalizationsTest.DONOTCHECKPREFIX + "ElectricalPotentialUnit.NanoV", SI_DERIVED,
66                          ElectricalPotentialUnit.VOLT, 1e-9);
67          assertTrue("Can create a new ElectricalPotentialUnit", null != myEPU);
68          checkUnitRatioNameAndAbbreviation(myEPU, 1e-9, 0.1, "!NanoVolt!", "!NanoV!");
69  
70          myEPU =
71                  new ElectricalPotentialUnit(PowerUnit.FOOT_POUND_FORCE_PER_HOUR, ElectricalCurrentUnit.MICROAMPERE,
72                          UnitLocalizationsTest.DONOTCHECKPREFIX + "ElectricalPotentialUnit.fpfph/microA",
73                          UnitLocalizationsTest.DONOTCHECKPREFIX + "ElectricalPotentialUnit.fpfph/uA", UnitSystem.IMPERIAL);
74          assertTrue("Can create a new ElectricalPotentialUnit", null != myEPU);
75          checkUnitRatioNameAndAbbreviation(myEPU, 376.6, 0.1, "!fpfph/microA!", "!fpfph/uA!");
76      }
