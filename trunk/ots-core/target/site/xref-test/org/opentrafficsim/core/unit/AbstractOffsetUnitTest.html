14  public class AbstractOffsetUnitTest<OU extends OffsetUnit<OU>> extends AbstractUnitTest<OU>
15  {
16      /**
17       * Verify one length conversion factor to standard unit and the localization of the name and abbreviation.
18       * @param ou Unit to check
19       * @param expectedRatio Double; expected ratio
20       * @param expectedOffset Double; expected offset
21       * @param precision Double; precision of verification
22       * @param expectedName String; expected name in the resources
23       * @param expectedAbbreviation String; expected abbreviation in the resources
24       */
25      protected final void checkUnitRatioOffsetNameAndAbbreviation(final OU ou, final double expectedRatio,
26              final double expectedOffset, final double precision, final String expectedName, final String expectedAbbreviation)
27      {
28          assertEquals(String.format("zero %s is about %f reference unit", ou.getNameKey(), expectedOffset), expectedOffset,
29                  ou.getOffsetToStandardUnit(), precision);
30          assertEquals(String.format("one %s is about %f reference unit", ou.getNameKey(), expectedRatio), expectedRatio,
31                  ou.getConversionFactorToStandardUnit(), precision);
32          assertEquals(String.format("Name of %s is %s", ou.getNameKey(), expectedName), expectedName, ou.getName());
33          assertEquals(String.format("Abbreviation of %s is %s", ou.getNameKey(), expectedAbbreviation), expectedAbbreviation,
34                  ou.getAbbreviation());
35      }
36  
37      /**
38       * @param fromUnit U; the unit to convert from
39       * @param toUnit U; the unit to convert to
40       * @return offset to convert a value from fromUnit to toUnit
41       */
42      public final double getOffsetTo(final OU fromUnit, final OU toUnit)
43      {
44          double fromOffset = fromUnit.getOffsetToStandardUnit();
45          double fromFactor = fromUnit.getConversionFactorToStandardUnit();
46          double inStandard = (0d - fromOffset) * fromFactor;
47          double toOffset = toUnit.getOffsetToStandardUnit();
48          double toFactor = toUnit.getConversionFactorToStandardUnit();
49          double inToUnit = inStandard / toFactor + toOffset;
50          return inToUnit;
51      }
