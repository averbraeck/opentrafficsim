16  public final class AvailableLocalizations
17  {
18      /**
19       * This class should never be instantiated.
20       */
21      private AvailableLocalizations()
22      {
23          // no instantiation
24      }
25  
26      /**
27       * Build a list of locale names that are available for a given prefix.
28       * @param prefix String; the prefix of the localization file names
29       * @param path String; the path to the resource directory to scan
30       * @return List&lt;String&gt;; the list of matching locale names (which <b>always</b> starts with "en", even though there
31       *         may not be such a localization file)
32       */
33      public static List<String> availableLocalizations(final String prefix, final String path)
34      {
35          final String tail = ".properties";
36          File[] propertiesFiles = new File(path).listFiles(new FileFilter()
37          {
38              @Override
39              public boolean accept(final File pathname)
40              {
41                  String name = pathname.getName();
42                  return name.endsWith(tail) && name.startsWith(prefix + "_");
43              }
44          });
45          List<String> locales = new ArrayList<String>();
46          locales.add("en");
47          for (File f : propertiesFiles)
48          {
49              String localeName = f.getName().substring(prefix.length() + 1);
50              localeName = localeName.substring(0, 2);
51              locales.add(localeName);
52          }
53          return locales;
54      }
