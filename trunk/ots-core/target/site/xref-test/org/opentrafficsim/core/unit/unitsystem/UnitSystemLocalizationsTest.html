20  public class UnitSystemLocalizationsTest
21  {
22      /**
23       * Check that all UnitSystems have valid a nameKey and a valid abbreviationKey and test those keys in all available
24       * localizations.
25       */
26      @Test
27      public final void checkDefinedUnitSystems()
28      {
29          List<UnitSystem> unitSystems = new ArrayList<UnitSystem>();
30          Field[] fields = UnitSystem.class.getDeclaredFields();
31          for (int i = 0; i < fields.length; i++)
32          {
33              // System.out.println("Field[" + i + "]: " + fields[i]);
34              try
35              {
36                  UnitSystem us = (UnitSystem) fields[i].get(null);
37                  // System.out.println("Prints like " + us);
38                  // System.out.println("nameKey: " + us.getNameKey());
39                  unitSystems.add(us);
40              }
41              catch (Exception e)
42              {
43                  // That was not a UnitSystem
44              }
45          }
46          ArrayList<String> errors = new ArrayList<String>();
47          for (String localeName : AvailableLocalizations.availableLocalizations("localeunitsystem",
48                  this.getClass().getResource("").getPath() + "../../../../../"))
49          {
50              for (UnitSystem us : unitSystems)
51              {
52                  String nameKey = us.getNameKey();
53                  assertTrue("nameKey is non null", null != nameKey);
54                  assertTrue("Name key must be non-empty", nameKey.length() > 0);
55                  String abbreviationKey = us.getAbbreviationKey();
56                  assertTrue("abbreviationKey is non null", null != abbreviationKey);
57                  assertTrue("Abbreviation key must be non-empty", abbreviationKey.length() > 0);
58                  String name = us.getName();
59                  String abbreviation = us.getAbbreviation();
60                  if (abbreviation.startsWith("!") && abbreviation.endsWith("!"))
61                  {
62                      errors.add(String.format("Missing translation for abbreviation %s to %s", abbreviationKey, localeName));
63                  }
64                  if (name.startsWith("!") && name.endsWith("!"))
65                  {
66                      errors.add(String.format("Missing translation for name %s to %s", nameKey, localeName));
67                  }
68              }
69          }
70          for (String s : errors)
71          {
72              System.out.println(s);
73          }
74          assertTrue("There should be no errors in UnitSystemLocalizations", errors.isEmpty());
75      }
76  }
