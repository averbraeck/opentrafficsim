19      /**
20       * Currently the parser throws an Error if something is not right. This behavior may be changed in the future. To prevent
21       * the need to rewrite all parser tests, this wrapper will catch those error and assume null when they occur.
22       * @param inputString String; the coefficientString to parse
23       * @param expectedResult String; the expected output
24       */
25      private static void parseString(final String inputString, final String expectedResult)
26      {
27          SICoefficients result = null;
28          try
29          {
30              result = new SICoefficients(SICoefficients.parse(inputString));
31          }
32          catch (UnitException e)
33          {
34              assertTrue("input \"" + inputString + "\" should have been parseable", null == expectedResult);
35              return;
36          }
37          assertTrue("input \"" + inputString + "\" should not have been parseable", null != expectedResult);
38          assertEquals("input \"" + inputString + "\" did not yield the expected result", expectedResult, result.toString());
39      }
40  
41      /**
42       * Test the create method.
43       */
44      @SuppressWarnings("static-method")
45      @Test
46      public final void parser()
47      {
48          parseString("", "1");
49          parseString("1", "1");
50          parseString("kgm/s2", "kg.m/s2");
51          parseString("kg-2m^3/s2/A", "m3/kg2/s2/A");
52          parseString("s-2", "1/s2");
53          parseString("cd/mol^2", "cd/mol2");
54          parseString("Kmmol3/AsKcd4", "m.s.K2.cd4.mol3/A");
55          parseString("m", "m");
56          parseString("mol", "mol");
57          parseString("OIhFUSAIFAHsufasfs", null);
58          parseString("/s", "1/s");
59          parseString("/s/s", "1/s2");
60          parseString("s-2345", "1/s2345");
61          parseString("s3/s3", "1");
62          parseString("s3/s4", "1/s");
63          parseString("s-3/s-4", "s");
64          parseString("sssss", "s5");
65          parseString("/s/s/s/s/s", "1/s5");
66          parseString("s^3", "s3");
67          parseString("s^-3", "1/s3");
68          parseString("/s^4", "1/s4");
69          parseString("/s^-5", "s5");
70          parseString("/", null);
71          parseString("/s/", null);
72      }

74      /**
75       * Execute one multiplication test.
76       * @param leftString String; coefficientString of first operand
77       * @param rightString String; coefficientString of second operand
78       * @param expectedResult String; coefficientString of the expected result
79       */
80      private static void multiplyTest(final String leftString, final String rightString, final String expectedResult)
81      {
82          try
83          {
84              SICoefficients left = new SICoefficients(SICoefficients.parse(leftString));
85              SICoefficients right = new SICoefficients(SICoefficients.parse(rightString));
86              SICoefficients product = SICoefficients.multiply(left, right);
87              assertEquals("input \"" + leftString + "\" times " + rightString + " did not yield the expected result",
88                      expectedResult, product.toString());
89          }
90          catch (UnitException exception)
91          {
92              fail("Caught unexpected UnitException");
93          }
94      }
95  
96      /**
97       * Test the multiply method.
98       */
99      @SuppressWarnings("static-method")
100     @Test
101     public final void multiply()
102     {
103         multiplyTest("1", "1", "1");
104         multiplyTest("A", "K", "A.K");
105         multiplyTest("A-1", "A", "1");
106         multiplyTest("A-2", "/A^2", "1/A4");
107         multiplyTest("Kmmol3/AsKcd4", "1", "m.s.K2.cd4.mol3/A");
108         multiplyTest("1", "Kmmol3/AsKcd4", "m.s.K2.cd4.mol3/A");
109         multiplyTest("Kmmol3", "/AsKcd4", "m.s.K2.cd4.mol3/A");
110         multiplyTest("kgm/s2", "m", "kg.m2/s2");
111     }

116      * @param rightString String; coefficientString of divisor
117      * @param expectedResult String; coefficientString of the expected result
118      */
119     private static void divideTest(final String leftString, final String rightString, final String expectedResult)
120     {
121         try

123             SICoefficients left = new SICoefficients(SICoefficients.parse(leftString));

132         }
133     }
