21  public class FrequencyUnitTest extends AbstractUnitTest<FrequencyUnit>
22  {
23      /**
24       * Set the locale to "en" so we know what texts should be retrieved from the resources.
25       */
26      @SuppressWarnings("static-method")
27      @Before
28      public final void setup()
29      {
30          DefaultLocale.setLocale(new Locale("en"));
31      }
32  
33      /**
34       * Verify the result of some get*Key methods.
35       */
36      @Test
37      public final void keys()
38      {
39          checkKeys(FrequencyUnit.HERTZ, "FrequencyUnit.Hertz", "FrequencyUnit.Hz");
40      }
41  
42      /**
43       * Verify conversion factors, English names and abbreviations.
44       */
45      @Test
46      public final void conversions()
47      {
48          checkUnitRatioNameAndAbbreviation(FrequencyUnit.HERTZ, 1, 0.000001, "Hertz", "Hz");
49          checkUnitRatioNameAndAbbreviation(FrequencyUnit.KILOHERTZ, 1000, 0.0001, "kilohertz", "kHz");
50          // Check two conversions between non-standard units
51          assertEquals("one KILOHERTZ is 0.001 MEGAHERTZ", 0.001,
52                  getMultiplicationFactorTo(FrequencyUnit.KILOHERTZ, FrequencyUnit.MEGAHERTZ), 0.00000000001);
53          assertEquals("one MEGAHERTZ is 1000 KILOHERTZ", 1000,
54                  getMultiplicationFactorTo(FrequencyUnit.MEGAHERTZ, FrequencyUnit.KILOHERTZ), 0.0005);
55          checkUnitRatioNameAndAbbreviation(FrequencyUnit.GIGAHERTZ, 1e9, 1e3, "gigahertz", "GHz");
56          checkUnitRatioNameAndAbbreviation(FrequencyUnit.TERAHERTZ, 1e12, 1e6, "terahertz", "THz");
57      }
58  
59      /**
60       * Verify that we can create our own Frequency unit.
61       */
62      @Test
63      public final void createFrequencyUnitt()
64      {
65          FrequencyUnit myFU =
66                  new FrequencyUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "FrequencyUnit.MiddleA",
67                          UnitLocalizationsTest.DONOTCHECKPREFIX + "FrequencyUnit.MA", OTHER, FrequencyUnit.KILOHERTZ, 0.440);
68          assertTrue("Can create a new ForceUnit", null != myFU);
69          checkUnitRatioNameAndAbbreviation(myFU, 440, 0.0001, "!MiddleA!", "!MA!");
70      }
