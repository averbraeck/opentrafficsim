19  public class AccelerationUnitTest extends AbstractUnitTest<AccelerationUnit>
20  {
21      /**
22       * Set the locale to "en" so we know what texts should be retrieved from the resources.
23       */
24      @SuppressWarnings("static-method")
25      @Before
26      public final void setup()
27      {
28          DefaultLocale.setLocale(new Locale("en"));
29      }
30  
31      /**
32       * Verify the result of some get*Key methods.
33       */
34      @Test
35      public final void keys()
36      {
37          checkKeys(AccelerationUnit.INCH_PER_SECOND_2, "AccelerationUnit.inch_per_second_squared", "AccelerationUnit.in/s^2");
38      }
39  
40      /**
41       * Verify conversion factors, English names and abbreviations.
42       */
43      @Test
44      public final void conversions()
45      {
46          checkUnitRatioNameAndAbbreviation(AccelerationUnit.METER_PER_SECOND_2, 1, 0.00000001, "meter per second squared",
47                  "m/s^2");
48          checkUnitRatioNameAndAbbreviation(AccelerationUnit.KM_PER_HOUR_2, 1 / 3.6 / 3600, 0.0005, "km per hour squared",
49                  "km/h^2");
50          checkUnitRatioNameAndAbbreviation(AccelerationUnit.FOOT_PER_SECOND_2, 0.3048, 0.00001, "foot per second squared",
51                  "ft/s^2");
52          // Check two conversions between non-standard units
53          assertEquals("one FOOT PER SECOND PER SECOND is ??? KM PER HOUR PER HOUR", 3950.208,
54                  getMultiplicationFactorTo(AccelerationUnit.FOOT_PER_SECOND_2, AccelerationUnit.KM_PER_HOUR_2), 0.01);
55          // Check conversion factor to standard unit for all remaining acceleration units
56          checkUnitRatioNameAndAbbreviation(AccelerationUnit.INCH_PER_SECOND_2, 0.0254, 0.0000000001, "inch per second squared",
57                  "in/s^2");
58          checkUnitRatioNameAndAbbreviation(AccelerationUnit.MILE_PER_HOUR_2, 0.000124177778, 0.0000000001,
59                  "mile per hour squared", "mi/h^2");
60          checkUnitRatioNameAndAbbreviation(AccelerationUnit.MILE_PER_SECOND_2, 1609.344, 0.000001, "mile per second squared",
61                  "mi/s^2");
62          checkUnitRatioNameAndAbbreviation(AccelerationUnit.KNOT_PER_SECOND, 0.514444444, 0.000001, "knot per second", "kt/s");
63          checkUnitRatioNameAndAbbreviation(AccelerationUnit.MILE_PER_HOUR_PER_SECOND, 0.44704, 0.00000001,
64                  "mile per hour per second", "mi/h/s");
65      }
66  
67      /**
68       * Check that g is about 9.8 m/s/s.
69       */
70      @Test
71      public final void gravityConstant()
72      {
73          checkUnitRatioNameAndAbbreviation(AccelerationUnit.STANDARD_GRAVITY, 9.8, 0.02, "standard gravity", "g");
74      }
75  }
