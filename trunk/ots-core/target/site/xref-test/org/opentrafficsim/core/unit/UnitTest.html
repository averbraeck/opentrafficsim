19      /**
20       * Test the lookupUnitWithSICoefficients method.
21       */
22      @SuppressWarnings("static-method")
23      @Test
24      public final void lookupUnitWithSICoefficients()
25      {
26          assertEquals("ABVOLT is expressed in Volt", "[V]",
27                  Unit.lookupUnitWithSICoefficients(ElectricalPotentialUnit.ABVOLT.getSICoefficients().toString())
28                          .toString());
29          assertEquals(
30                  "ABVOLT / STATAMPERE is expressed in Ohm",
31                  "[\u03A9]",
32                  Unit.lookupUnitWithSICoefficients(
33                          SICoefficients.divide(ElectricalPotentialUnit.ABVOLT.getSICoefficients(),
34                                  ElectricalCurrentUnit.STATAMPERE.getSICoefficients()).toString()).toString());
35          assertEquals(
36                  "ABVOLT * STATAMPERE is expressed in Watt",
37                  "[W]",
38                  Unit.lookupUnitWithSICoefficients(
39                          SICoefficients.multiply(ElectricalPotentialUnit.ABVOLT.getSICoefficients(),
40                                  ElectricalCurrentUnit.STATAMPERE.getSICoefficients()).toString()).toString());
41          assertEquals(
42                  "ABVOLT / Watt is expressed in Ohm",
43                  "[!1/A!]",
44                  Unit.lookupOrCreateUnitWithSICoefficients(
45                          SICoefficients.divide(ElectricalPotentialUnit.ABVOLT.getSICoefficients(),
46                                  PowerUnit.WATT.getSICoefficients()).toString()).toString());
47          assertEquals(
48                  "ABVOLT * KILOVOLT is expressed in kg2.m4/s6/A2",
49                  "[!kg2.m4/s6/A2!]",
50                  Unit.lookupOrCreateUnitWithSICoefficients(
51                          SICoefficients.multiply(ElectricalPotentialUnit.ABVOLT.getSICoefficients(),
52                                  ElectricalPotentialUnit.KILOVOLT.getSICoefficients()).toString()).toString());
53      }

55      /**
56       * Check objects returned by getAllUnitsOfType.
57       */
58      @SuppressWarnings("static-method")
59      @Test
60      public final void getAllUnitsOfType()
61      {
62          Unit<?>[] baseUnits =
63                  {MassUnit.KILOGRAM, LengthUnit.METER, ElectricalCurrentUnit.AMPERE, TimeUnit.SECOND,
64                          TemperatureUnit.KELVIN, /* LuminousIntencity . CANDELA , ? ? ? . mol */};
65          for (Unit<?> u : baseUnits)
66          {
67              for (Object unitObject : u.getAllUnitsOfThisType())
68              {
69                  assertTrue("getAllUnitsOfThisType returns Units", unitObject instanceof Unit);
70                  Unit<?> u2 = (Unit<?>) unitObject;
71                  assertEquals("Standard unit of " + u2 + " should be " + u, u, u2.getStandardUnit());
72              }
73          }
74      }
