40          assertEquals("Value in Fahrenheit", 100.4f, temperatureFS.getValueInUnit(TemperatureUnit.DEGREE_FAHRENHEIT),

44          // assertEquals("Unit should now be Fahrenheit", TemperatureUnit.DEGREE_FAHRENHEIT, temperatureFS.getUnit());
45          // assertEquals("Value in unit is now the equivalent in Fahrenheit", 100.4f, temperatureFS.getValueInUnit(),

48          assertEquals("Value in Fahrenheit", 100.4f, temperatureFS.getValueInUnit(TemperatureUnit.DEGREE_FAHRENHEIT),
49                  0.1);
50          assertTrue("Value is absolute", temperatureFS.isAbsolute());
51          assertFalse("Value is absolute", temperatureFS.isRelative());
52          assertEquals("Unit of copy made before calling setUnit should be unchanged", tempUnit, u2.getUnit());
53  
54          LengthUnit lengthUnit = LengthUnit.INCH;
55          value = 12f;
56          FloatScalar.Abs<LengthUnit> lengthFS = new FloatScalar.Abs<LengthUnit>(value, lengthUnit);
57          // System.out.println("lengthFS is " + lengthFS);
58          assertEquals("Unit should be Inch", lengthUnit, lengthFS.getUnit());
59          assertEquals("Value is what we put in", value, lengthFS.getValueInUnit(), 0.0001);
60          assertEquals("Value in SI is equivalent in Meter", 0.3048f, lengthFS.getValueSI(), 0.0005);
61          assertEquals("Value in Foot", 1f, lengthFS.getValueInUnit(LengthUnit.FOOT), 0.0001);
62          FloatScalar.Abs<LengthUnit> copy = lengthFS.copy();
63          assertEquals("compareTo should return 0", 0, copy.compareTo(lengthFS));
64          FloatScalar.Abs<LengthUnit> bigger =
65                  new FloatScalar.Abs<LengthUnit>(lengthFS.getValueInUnit() + 100, lengthFS.getUnit());
66          assertEquals("compareTo should return 1", 1, bigger.compareTo(lengthFS));
67          assertEquals("compareTo should return -1", -1, lengthFS.compareTo(bigger));
68          assertEquals("Unit of copy should still be in Inch", LengthUnit.INCH, copy.getUnit());
69          assertTrue("copy should be equal to original", copy.equals(lengthFS));
70          assertEquals("copy should have same hashcode as original", copy.hashCode(), lengthFS.hashCode());
71          assertTrue("original should be equal to copy", lengthFS.equals(copy));
72          assertTrue("original should be equal to itself", lengthFS.equals(lengthFS));
73  
74          FloatScalar.Abs<LengthUnit> scalarFromScalar = new FloatScalar.Abs<LengthUnit>(lengthFS);
75          assertEquals("Unit should be Inch", lengthUnit, scalarFromScalar.getUnit());
76          assertEquals("Value is what we put in", value, scalarFromScalar.getValueInUnit(), 0.0001);
77          assertEquals("Value in SI is equivalent in Meter", 0.3048f, scalarFromScalar.getValueSI(), 0.0005);
78          assertEquals("Value in Foot", 1f, scalarFromScalar.getValueInUnit(LengthUnit.FOOT), 0.0001);
79  
80          MutableFloatScalar.Abs<LengthUnit> m = lengthFS.mutable();
81          assertEquals("Unit should be Inch", lengthUnit, m.getUnit());
82          assertEquals("Value is what we put in", value, m.getValueInUnit(), 0.0001);
83          assertEquals("Value in SI is equivalent in Meter", 0.3048f, m.getValueSI(), 0.0005);
84          assertEquals("Value in Foot", 1f, m.getValueInUnit(LengthUnit.FOOT), 0.0001);
85          assertTrue("mutable version should be equal to original", m.equals(lengthFS));
86          assertEquals("mutable version should have same hashCode as original", m.hashCode(), lengthFS.hashCode());
87          MutableFloatScalar.Abs<LengthUnit> mm = m.copy();
88          FloatScalar.Abs<LengthUnit> imm = m.immutable();
89          assertEquals("compareTo should return 0", 0, m.compareTo(mm));
90          assertEquals("compareTo should return 0", 0, mm.compareTo(m));
91          // Modify the mutable
92          m.setSI(m.getValueSI() + 1);
93          assertFalse("modified version should NOT be equal to original", m.equals(lengthFS));
94          assertTrue("HashCode of modified version should not be equal to hashCode of original",
95                  m.hashCode() != lengthFS.hashCode());
96          assertEquals("original should have the original value", value, lengthFS.getValueInUnit(), 0.0001);
97          assertEquals("copy of mutable version should have the original value", value, mm.getValueInUnit(), 0.0001);
98          assertEquals("immutable variant of mutable version should have the original value", value,
99                  imm.getValueInUnit(), 0.0001);
100         assertEquals("compareTo should return 1", 1, m.compareTo(mm));
101         assertEquals("compareTo should return -1", -1, mm.compareTo(m));
102         // undo change
103         m.setSI(lengthFS.getValueSI());
104         assertTrue("restored mutable version should be equal to original", m.equals(lengthFS));
105         assertEquals("restored mutable version should have same hashCode as original", m.hashCode(),
106                 lengthFS.hashCode());
107         FloatScalar.Abs<LengthUnit> differentUnit =
108                 new FloatScalar.Abs<LengthUnit>(lengthFS.getValueSI(), LengthUnit.METER);
109         assertTrue("floatScalar with different unit, but same SI value and also absolute should be equal",
110                 lengthFS.equals(differentUnit));
111         assertEquals("floatScalar with different unit, but same SI value should have same hashCode",
112                 lengthFS.hashCode(), differentUnit.hashCode());
113         FloatScalar.Rel<LengthUnit> differentUnitRel =
114                 new FloatScalar.Rel<LengthUnit>(lengthFS.getValueSI(), LengthUnit.METER);
115         assertFalse("floatScalar with different unit, but same SI value but not also absolute should NOT be equal",
116                 lengthFS.equals(differentUnitRel));
117         assertFalse("floatScalar with different unit, but same SI value but not also absolute should NOT be equal",
118                 differentUnitRel.equals(lengthFS));
119 
120         assertEquals("intValue should return rounded value in SI", Math.round(lengthFS.getValueSI()),
121                 lengthFS.intValue(), 0.0001);
122         assertEquals("longValue should return rounded value in SI", Math.round(lengthFS.getValueSI()),
123                 lengthFS.longValue(), 0.0001);
124         assertEquals("floatValue should return rounded value in SI", lengthFS.getValueSI(), lengthFS.floatValue(),
125                 0.0001);
126         assertEquals("doubleValue should return rounded value in SI", lengthFS.getValueSI(), lengthFS.doubleValue(),
127                 0.0001);
128         assertFalse("equals to null should return false", lengthFS.equals(null));

327     /**
328      * Test plus, minus, times.
329      */
330     @SuppressWarnings("static-method")
331     @Test
332     public final void simpleArithmetic()
333     {
334         float leftValue = 123.456f;
335         float rightValue = 21.098f;
336         float rightValue2 = 1.0987f;
337         FloatScalar.Abs<LengthUnit> leftAbs = new FloatScalar.Abs<LengthUnit>(leftValue, LengthUnit.METER);
338         FloatScalar.Rel<LengthUnit> right = new FloatScalar.Rel<LengthUnit>(rightValue, LengthUnit.INCH);
339         FloatScalar.Rel<LengthUnit> right2 = new FloatScalar.Rel<LengthUnit>(rightValue2, LengthUnit.MILLIMETER);
340         MutableFloatScalar.Abs<LengthUnit> sum = MutableFloatScalar.plus(leftAbs, right);
341         assertEquals("result should be in METER", LengthUnit.METER, sum.getUnit());
342         assertEquals("value of result should be sum of meter equivalent of values", leftValue + rightValue * 0.0254,
343                 sum.getValueSI(), 0.0001);
344         FloatScalar.Rel<LengthUnit> leftRel = new FloatScalar.Rel<LengthUnit>(leftValue, LengthUnit.METER);
345         MutableFloatScalar.Rel<LengthUnit> sum2 =
346                 MutableFloatScalar.plus(LengthUnit.MILLIMETER, leftRel, right, right2);
347         assertEquals("result should be in MILLIMETER", LengthUnit.MILLIMETER, sum2.getUnit());
348         assertEquals("value in SI should be sum of meter equivalent of values", leftValue + rightValue * 0.0254
349                 + rightValue2 / 1000, sum2.getValueSI(), 0.0001);
350         assertEquals("value in \"own\" unit should be equivalent in MILLIMETER", 1000
351                 * (leftValue + rightValue * 0.0254) + rightValue2, sum2.getValueInUnit(), 0.1);
352         MutableFloatScalar.Abs<LengthUnit> difference = MutableFloatScalar.minus(leftAbs, right, right2);
353         assertEquals("result should be in METER", LengthUnit.METER, difference.getUnit());
354         assertEquals("value in SI should be sum of meter equivalent of values", leftValue - rightValue * 0.0254
355                 - rightValue2 / 1000, difference.getValueSI(), 0.0001);
356         assertEquals("value in \"own\" unit should be equivalent in METER", leftValue - rightValue * 0.0254
357                 - rightValue2 / 1000, difference.getValueInUnit(), 0.1);
358         MutableFloatScalar.Rel<LengthUnit> differenceRel = MutableFloatScalar.minus(leftRel, right, right2);
359         assertEquals("result should be in METER", LengthUnit.METER, differenceRel.getUnit());
360         assertEquals("value in SI should be sum of meter equivalent of values", leftValue - rightValue * 0.0254
361                 - rightValue2 / 1000, differenceRel.getValueSI(), 0.0001);
362         assertEquals("value in \"own\" unit should be equivalent in METER", leftValue - rightValue * 0.0254
363                 - rightValue2 / 1000, differenceRel.getValueInUnit(), 0.001);
364         differenceRel = MutableFloatScalar.minus(sum.immutable(), leftAbs);
365         assertEquals("result should be in METER", LengthUnit.METER, difference.getUnit());
366         assertEquals("value of result should be minus leftValue", rightValue * 0.0254, differenceRel.getValueSI(),
367                 0.0001);
368         differenceRel =
369                 MutableFloatScalar.minus(new FloatScalar.Abs<LengthUnit>(1, LengthUnit.FOOT),
370                         new FloatScalar.Abs<LengthUnit>(1, LengthUnit.INCH));
371         assertEquals("result should be 11 inches", 11 * 0.0254, differenceRel.getValueSI(), 0.0001);
372         MutableFloatScalar.Abs<?> surface = MutableFloatScalar.multiply(leftAbs, difference.immutable());
373         // System.out.println("surface is " + surface);
374         assertEquals("Surface should be in square meter", AreaUnit.SQUARE_METER.getSICoefficientsString(), surface
375                 .getUnit().getSICoefficientsString());
376         assertEquals("Surface should be equal to the product of contributing values",
377                 leftAbs.getValueSI() * difference.getValueSI(), surface.getValueSI(), 0.05);
378         MutableFloatScalar.Rel<?> relSurface = MutableFloatScalar.multiply(right, right2);
379         assertEquals("Surface should be in square meter", AreaUnit.SQUARE_METER.getSICoefficientsString(), relSurface
380                 .getUnit().getSICoefficientsString());
381         assertEquals("Surface should be equal to the product of contributing values",
382                 right.getValueSI() * right2.getValueSI(), relSurface.getValueSI(), 0.00000005);
383         assertTrue("FloatScalar should be equal to itself", leftAbs.equals(leftAbs));
384         FloatScalar<?> copy = new FloatScalar.Abs<LengthUnit>(leftAbs);
385         assertTrue("Copy of FloatScalar should be equal to itself", leftAbs.equals(copy));
386         copy = new FloatScalar.Rel<LengthUnit>(leftAbs.getValueInUnit(), leftAbs.getUnit());
387         assertTrue("this copy should be relative", copy instanceof Relative);
388         assertFalse("Relative can not be equal to (otherwise equal) absolute", leftAbs.equals(copy));
389         assertFalse("FloatScalar is not a String", copy.equals("String"));
390         FloatScalar.Abs<TimeUnit> timeScalar = new FloatScalar.Abs<TimeUnit>(leftValue, TimeUnit.SECOND);
391         assertEquals("Value should match", leftAbs.getValueSI(), timeScalar.getValueSI(), 0.0001);
392         assertTrue("Both are absolute", leftAbs.isAbsolute() && timeScalar.isAbsolute());
393         assertFalse("Absolute length is not equal to absolute time with same value", leftAbs.equals(timeScalar));
394         MutableFloatScalar<LengthUnit> left = new MutableFloatScalar.Abs<LengthUnit>(leftValue, LengthUnit.METER);
395         left.add(right);
396         assertEquals("after add-and-becomes the type should not be changed", LengthUnit.METER, left.getUnit());
397         assertEquals("after add-and-becomes the value should be changed", leftValue + rightValue * 0.0254,
398                 left.getValueSI(), 0.0001);
399         left = new MutableFloatScalar.Abs<LengthUnit>(leftValue, LengthUnit.METER);
400         left.subtract(right);
401         assertEquals("after subtract-and-becomes the type should not be changed", LengthUnit.METER, left.getUnit());
402         assertEquals("after subtract-and-becomes the value should be changed", leftValue - rightValue * 0.0254,
403                 left.getValueSI(), 0.0001);
404     }

411     public final void divide()
412     {
413         float leftValue = 123.456f;

415         FloatScalar.Abs<LengthUnit> leftAbs = new FloatScalar.Abs<LengthUnit>(leftValue, LengthUnit.METER);

417         MutableFloatScalar.Abs<SIUnit> ratio = MutableFloatScalar.divide(leftAbs, rightAbs);
418         String unitString = ratio.getUnit().getAbbreviation();
419         if (unitString.endsWith("!"))
420         {
421             unitString = unitString.substring(0, unitString.length() - 1);
422         }
423         if (unitString.startsWith("!"))
424         {
425             unitString = unitString.substring(1);
426         }
427         assertEquals("result should be in SECOND", TimeUnit.SECOND.toString(), unitString);
428         assertEquals("value in SI should be ratio of SI equivalent of values", leftValue / rightValue * 3.600,
429                 ratio.getValueSI(), 0.0001);
430         FloatScalar.Rel<LengthUnit> leftRel = new FloatScalar.Rel<LengthUnit>(leftValue, LengthUnit.METER);
431         FloatScalar.Rel<SpeedUnit> rightRel = new FloatScalar.Rel<SpeedUnit>(rightValue, SpeedUnit.KM_PER_HOUR);
432         MutableFloatScalar.Rel<SIUnit> ratioRel = MutableFloatScalar.divide(leftRel, rightRel);
433         unitString = ratioRel.getUnit().getAbbreviation();
434         if (unitString.endsWith("!"))
435         {
436             unitString = unitString.substring(0, unitString.length() - 1);
437         }
438         if (unitString.startsWith("!"))
439         {
440             unitString = unitString.substring(1);
441         }
442         assertEquals("result should be in SECOND", TimeUnit.SECOND.toString(), unitString);
443         assertEquals("value in SI should be ratio of SI equivalent of values", leftValue / rightValue * 3.600,
444                 ratioRel.getValueSI(), 0.0001);
