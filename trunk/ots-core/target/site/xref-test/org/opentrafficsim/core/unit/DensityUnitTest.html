21  public class DensityUnitTest extends AbstractUnitTest<DensityUnit>
22  {
23      /**
24       * Set the locale to "en" so we know what texts should be retrieved from the resources.
25       */
26      @SuppressWarnings("static-method")
27      @Before
28      public final void setup()
29      {
30          DefaultLocale.setLocale(new Locale("en"));
31      }
32  
33      /**
34       * Verify the result of some get*Key methods.
35       */
36      @Test
37      public final void keys()
38      {
39          checkKeys(DensityUnit.KG_PER_METER_3, "DensityUnit.kilogram_per_cubic_meter", "DensityUnit.kg/m^3");
40      }
41  
42      /**
43       * Verify conversion factors, English names and abbreviations.
44       */
45      @Test
46      public final void conversions()
47      {
48          checkUnitRatioNameAndAbbreviation(DensityUnit.KG_PER_METER_3, 1, 0.00000001, "kilogram per cubic meter", "kg/m^3");
49          checkUnitRatioNameAndAbbreviation(DensityUnit.GRAM_PER_CENTIMETER_3, 1000, 0.0001, "gram per cubic centimeter",
50                  "g/cm^3");
51          // Check two conversions between two units
52          assertEquals("one KG PER CUBIC METER is 0.0001 GRAM PER CUBIC CENTIMETER", 0.001,
53                  getMultiplicationFactorTo(DensityUnit.KG_PER_METER_3, DensityUnit.GRAM_PER_CENTIMETER_3), 0.000000001);
54      }
55  
56      /**
57       * Verify that we can create our own density unit.
58       */
59      @Test
60      public final void createDensityUnit()
61      {
62          DensityUnit myDU =
63                  new DensityUnit(UnitLocalizationsTest.DONOTCHECKPREFIX + "DensityUnit.DensityUnit",
64                          UnitLocalizationsTest.DONOTCHECKPREFIX + "DensityUnit.SPCF", SI_DERIVED, DensityUnit.KG_PER_METER_3,
65                          515.317882);
66          assertTrue("Can create a new DensityUnit", null != myDU);
67          checkUnitRatioNameAndAbbreviation(myDU, 515.3, 0.1, "!DensityUnit!", "!SPCF!");
68      }
